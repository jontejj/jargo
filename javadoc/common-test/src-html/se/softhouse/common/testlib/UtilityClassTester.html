015package se.softhouse.common.testlib;

017import com.google.common.base.Function;
018import com.google.common.base.Predicate;
019import com.google.common.collect.FluentIterable;
020import com.google.common.collect.ImmutableSet;
021import com.google.common.reflect.ClassPath;
022import com.google.common.reflect.ClassPath.ClassInfo;
023
024import javax.annotation.concurrent.Immutable;
025import java.io.IOException;
026import java.lang.reflect.Constructor;
027import java.lang.reflect.InvocationTargetException;
028import java.lang.reflect.Method;
029import java.lang.reflect.Modifier;
030import java.util.Arrays;
031
032import static org.fest.assertions.Assertions.*;
033import static se.softhouse.common.testlib.ReflectionUtil.hasInstanceFields;
034import static se.softhouse.common.testlib.ReflectionUtil.isStatic;
035
036/**
037 * The reasoning behind testing code that doesn't do anything is to achieve 100%
038 * code coverage and to notice when the code
039 * coverage drops. Otherwise one could always think, hey I don't have 100% code
040 * coverage anymore but it's PROBABLY because of my private constructors
041 * (or any other code that's not used but preferred to have).
042 * This makes it easy to spot untested methods without having to check that it
043 * only was a private constructor etc.
044 */
045@Immutable
046public final class UtilityClassTester
047{
048        private UtilityClassTester()
049        {
050        }
051
052        /**
053         * Detects if a utility class isn't final or if its no-args constructor
054         * isn't private. Also calls the constructor to get code coverage for it.
055         * The constructor is allowed to throw exceptions.
056         */
057        public static void testUtilityClassDesign(Class<?> ... utilityClasses)
058        {
059                testUtilityClassDesign(Arrays.asList(utilityClasses));
060        }
061
062        /**
063         * {@link Iterable} version of {@link #testUtilityClassDesign(Class...)}.
064         */
065        public static void testUtilityClassDesign(Iterable<Class<?>> utilityClasses)
066        {
067                for(Class<?> clazz : utilityClasses)
068                {
069                        try
070                        {
071                                Constructor<?> constructor = clazz.getDeclaredConstructor();
072                                assertThat(constructor.getModifiers() & Modifier.PRIVATE).as("Constructor for " + clazz + " should be private.")
073                                                .isEqualTo(Modifier.PRIVATE);
074                                constructor.setAccessible(true);
075                                assertThat(constructor.newInstance()).isNotNull();
076                                assertThat(clazz.getModifiers() & Modifier.FINAL).as("Utility class " + clazz + " not final").isEqualTo(Modifier.FINAL);
077                        }
078                        catch(NoSuchMethodException e)
079                        {
080                                throw new IllegalArgumentException("No no-arg constructor found for " + clazz, e);
081                        }
082                        catch(InvocationTargetException e)
083                        {
084                                // Some utility classes like to throw from their private constructor just to be on
085                                // the safe side, let's accept that
086                        }
087                        catch(InstantiationException e)
088                        {
089                                throw new IllegalArgumentException("Utility " + clazz + " may not be abstract", e);
090                        }
091                        catch(IllegalAccessException e)
092                        {
093                                // Let's just say that if we're not able to create an instance it's fine, after all
094                                // that's what we're testing here
095                        }
096                }
097        }
098
099        /**
100         * Goes through all {@link ClassLoader#loadClass(String) loadable} classes around {@code klazz}
101         * (and sub packages) and makes sure that the class is marked as final and that its constructor
102         * is private if all publicly available methods are static (by using
103         * {@link #testUtilityClassDesign(Iterable)}).<br>
104         * <b>Note:</b> Nested classes are not tested (yet)
105         * 
106         * @throws IOException if the attempt to read class path resources (jar files or directories)
107         *             failed.
108         * @throws IllegalArgumentException if no utility classes were found
109         */
110        public static void testUtilityClassDesignForAllClassesAround(Class<?> klazz) throws IOException
111        {
112                String packageName = klazz.getPackage().getName();
113                ImmutableSet<ClassInfo> classes = ClassPath.from(klazz.getClassLoader()).getTopLevelClassesRecursive(packageName);
114                Iterable<Class<?>> utilityClasses = FluentIterable.from(classes).transform(loadClasses()).filter(lookingLikeAUtilityClass()).toList();
115
116                assertThat(utilityClasses).as("No utitlity classes exists in " + packageName).isNotEmpty();
117                testUtilityClassDesign(utilityClasses);
118        }
119
120        private static Predicate<Class<?>> lookingLikeAUtilityClass()
121        {
122                return input ->
123                {
124                        for(Method method : input.getDeclaredMethods())
125                        {
126                                if(method.isSynthetic())// Don't count injected code
127                                {
128                                        continue;
129                                }
130                                if(!isStatic(method))
131                                        return false;
132                        }
133                        if(input.isInterface())
134                                return false;
135                        if(hasInstanceFields(input))
136                                return false;
137                        return true;
138                };
139        }
140
141        private static Function<ClassInfo, Class<?>> loadClasses()
142        {
143                return ClassInfo::load;
144        }
145}
