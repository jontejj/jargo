013package se.softhouse.common.testlib;
014
015import java.io.InputStream;
016import java.io.InputStreamReader;
017import java.util.concurrent.Callable;
018import java.util.concurrent.Future;
019
020import javax.annotation.concurrent.Immutable;
021
022import com.google.common.base.Charsets;
023import com.google.common.io.CharStreams;
024import com.google.common.util.concurrent.ListenableFuture;
025
026/**
027 * Utility for working with {@link Readable} and {@link InputStream stream} instances.
028 */
029@Immutable
030public final class Streams
031{
032        private Streams()
033        {
034        }
035
036        /**
037         * Reads from {@code source} asynchronously. Use {@link Future#get()} to get the
038         * characters read. Characters will be read using {@link Charsets#UTF_8 UTF-8}. A typical use
039         * case is to avoid <a
040         * href="http://www.javaworld.com/javaworld/jw-12-2000/jw-1229-traps.html?page=4">deadlock
041         * pitfalls</a> when launching {@link Process processes}.
042         */
043        public static ListenableFuture<String> readAsynchronously(InputStream source)
044        {
045                final Readable sourceInUTF8 = new InputStreamReader(source, Charsets.UTF_8);
046                Callable<String> reader = new Callable<String>(){
047                        @Override
048                        public String call() throws Exception
049                        {
050                                StringBuilder result = new StringBuilder();
051                                CharStreams.copy(sourceInUTF8, result);
052                                return result.toString();
053                        }
054                };
055                return Concurrently.run(reader);
056        }
057}
