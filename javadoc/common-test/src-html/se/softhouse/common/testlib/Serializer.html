013package se.softhouse.common.testlib;
014
015import static com.google.common.base.Preconditions.checkNotNull;
016
017import java.io.ByteArrayInputStream;
018import java.io.ByteArrayOutputStream;
019import java.io.IOException;
020import java.io.ObjectInputStream;
021import java.io.ObjectOutputStream;
022import java.io.Serializable;
023
024import javax.annotation.concurrent.Immutable;
025
026/**
027 * Aids tests that needs to test how APIs behave when their objects are serialized
028 */
029@Immutable
030public final class Serializer
031{
032        private Serializer()
033        {
034        }
035
036        /**
037         * Serializes {@code objectToSerialize} and reads the serialized bytes into an object again.
038         * Requires that readReplace() isn't overridden to return a type not assignable as a {@code T},
039         * for default implementations of {@link Serializable} this is true.
040         */
041        public static <T extends Serializable> T clone(T objectToSerialize)
042        {
043                checkNotNull(objectToSerialize);
044                try
045                {
046                        ByteArrayOutputStream baos = new ByteArrayOutputStream();
047                        ObjectOutputStream oos = new ObjectOutputStream(baos);
048                        oos.writeObject(objectToSerialize);
049                        ByteArrayInputStream bais = new ByteArrayInputStream(baos.toByteArray());
050                        ObjectInputStream ois = new ObjectInputStream(bais);
051
052                        @SuppressWarnings("unchecked")
053                        T deserializedObject = (T) ois.readObject();
054                        return deserializedObject;
055                }
056                catch(IOException e)
057                {
058                        throw new IllegalArgumentException("Failed to clone: " + objectToSerialize, e);
059                }
060                catch(ClassNotFoundException e)
061                {
062                        throw new IllegalArgumentException("Failed to clone: " + objectToSerialize, e);
063                }
064        }
065}
