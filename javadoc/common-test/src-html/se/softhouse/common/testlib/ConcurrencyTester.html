013package se.softhouse.common.testlib;
014
015import static com.google.common.base.Preconditions.checkNotNull;
016import static org.fest.assertions.Assertions.assertThat;
017
018import java.util.List;
019import java.util.concurrent.CountDownLatch;
020import java.util.concurrent.CyclicBarrier;
021import java.util.concurrent.ExecutorService;
022import java.util.concurrent.Executors;
023import java.util.concurrent.TimeUnit;
024import java.util.concurrent.atomic.AtomicReference;
025
026import javax.annotation.concurrent.Immutable;
027
028import com.google.common.util.concurrent.Atomics;
029
030/**
031 * Helps to test how code works when it's concurrently accessed.
032 */
033@Immutable
034public final class ConcurrencyTester
035{
036        private ConcurrencyTester()
037        {
038        }
039
040        /**
041         * Can {@link #create(int) create} unique {@link Runnable}s based on a number. Used as input to
042         * {@link ConcurrencyTester#verify(RunnableFactory, long, TimeUnit)}.
043         */
044        public interface RunnableFactory
045        {
046                /**
047                 * Number of times to run each of the {@link #create(int) created} {@link Runnable}s.
048                 */
049                int iterationCount();
050
051                /**
052                 * Creates a Runnable that should be run {@link #iterationCount()} number of times.
053                 * This method is called {@link ConcurrencyTester#NR_OF_CONCURRENT_RUNNERS} times so the
054                 * total number of executions will be {@code iteratationCount * NR_OF_CONCURRENT_RUNNERS}.
055                 * Any {@link RuntimeException} or {@link Error} thrown from the created runnable will be
056                 * caught and propagated through
057                 * {@link ConcurrencyTester#verify(RunnableFactory, long, TimeUnit)}.
058                 * For performance and test-harness reasons the variables created based on
059                 * {@code uniqueNumber} should be saved and reused by the repeated runs.
060                 * 
061                 * @param uniqueNumber
062                 *            should be used to differentiate results made from different threads to
063                 *            increase the odds of detecting thread-safety issues.
064                 */
065                Runnable create(int uniqueNumber);
066        }
067
068        /**
069         * The threads should have to fight for CPU time
070         */
071        private static final int RUNNERS_PER_PROCESSOR = 3;
072
073        /**
074         * A suitable thread count to have alive at the same time to cause some intended contention.
075         */
076        public static final int NR_OF_CONCURRENT_RUNNERS = Runtime.getRuntime().availableProcessors() * RUNNERS_PER_PROCESSOR;
077
078        /**
079         * Verifies that {@link Runnable}s created with {@code factory} can be run concurrently.
080         * Waits for the whole execution to finish for {@code timeout} in {@code unit} time.
081         * 
082         * @throws Throwable if any errors occurs during the concurrent executions
083         */
084        public static void verify(RunnableFactory factory, long timeout, TimeUnit unit) throws Throwable
085        {
086                /**
087                 * Makes sure that all threads are alive at the same time
088                 */
089                CyclicBarrier startSignal = new CyclicBarrier(NR_OF_CONCURRENT_RUNNERS);
090
091                CountDownLatch activeWorkers = new CountDownLatch(NR_OF_CONCURRENT_RUNNERS);
092
093                /**
094                 * Used by other threads to report failure
095                 */
096                AtomicReference<Throwable> failureReporter = Atomics.newReference();
097
098                ExecutorService executor = Executors.newFixedThreadPool(NR_OF_CONCURRENT_RUNNERS);
099                int iterationCount = factory.iterationCount();
100                for(int i = 0; i < NR_OF_CONCURRENT_RUNNERS; i++)
101                {
102                        Runnable codeToTest = checkNotNull(factory.create(i));
103                        executor.execute(new BarrieredRunnable(codeToTest, iterationCount, startSignal, activeWorkers, failureReporter));
104                }
105
106                InterruptedException interrupted = null;
107                try
108                {
109                        if(!activeWorkers.await(timeout, unit))
110                                throw new AssertionError(
111                                                activeWorkers.getCount() + " of " + NR_OF_CONCURRENT_RUNNERS + " did not finish within " + timeout + " " + unit);
112                }
113                catch(InterruptedException e)
114                {
115                        // Makes this method reentrant from the same thread
116                        // Otherwise there could be a risk that await would throw
117                        // InterruptedException again without actually running any code again
118                        Thread.interrupted();
119                        interrupted = e;
120                }
121                List<Runnable> leftoverTasks = executor.shutdownNow();
122                if(failureReporter.get() != null)
123                        throw failureReporter.get();
124                if(interrupted != null)
125                        // We were interrupted while verifying, propagate so that tests finish up quickly
126                        throw interrupted;
127                assertThat(leftoverTasks).as("Tasks remained even though activeWorkers reached zero").isEmpty();
128        }
129
130        /**
131         * Runs a {@link Runnable} ({@code iterationCount} times) after a {@code startSignal} has been
132         * given. When errors are encountered they are reported to {@code failureReporter} and the
133         * thread that created this instance is then {@link Thread#interrupt() interrupted}.
134         */
135        private static final class BarrieredRunnable implements Runnable
136        {
137                private final Thread originThread;
138                private final Runnable target;
139                private final int iterationCount;
140                private final CyclicBarrier startSignal;
141                private final AtomicReference<Throwable> failureReporter;
142                private final CountDownLatch activeWorkers;
143
144                private BarrieredRunnable(Runnable target, int iterationCount, CyclicBarrier startSignal, CountDownLatch activeWorkers,
145                                AtomicReference<Throwable> failureReporter)
146                {
147                        this.originThread = Thread.currentThread();
148                        this.target = target;
149                        this.iterationCount = iterationCount;
150                        this.startSignal = startSignal;
151                        this.activeWorkers = activeWorkers;
152                        this.failureReporter = failureReporter;
153                }
154
155                @Override
156                public void run()
157                {
158                        try
159                        {
160                                // Give all threads at most 10 seconds to come alive
161                                startSignal.await(10, TimeUnit.SECONDS);
162                                for(int i = 0; i < iterationCount; i++)
163                                {
164                                        target.run();
165                                        startSignal.await();
166                                }
167                        }
168                        catch(Throwable e)
169                        {
170                                // Don't report secondary failures
171                                if(failureReporter.compareAndSet(null, e))
172                                {
173                                        originThread.interrupt();
174                                }
175                                return;
176                        }
177                        activeWorkers.countDown();
178                }
179        }
180}
