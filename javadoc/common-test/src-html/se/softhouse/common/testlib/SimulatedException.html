013package se.softhouse.common.testlib;
014
015import org.fest.assertions.Fail;
016
017/**
018 * Used to simulate an exception. Commonly used to avoid throwing something like
019 * {@link IllegalStateException} or {@link IllegalArgumentException} in tests (catching those could
020 * mask other errors if the message on the exception isn't verified, and asserting that message
021 * would only cause further confusion as that's not really what the test is about). Catching
022 * {@link SimulatedException} should be safe in tests as that's the purpose of the test, to catch it
023 * and verify behavior when it occurs. <br>
024 * <b>Reminder:</b> don't forget to call {@link Fail#fail(String) fail} if the exception doesn't
025 * happen (otherwise you won't notice any test failures and a kitten may die)
026 */
027public class SimulatedException extends RuntimeException
028{
029        private static final long serialVersionUID = 1L;
030
031        /**
032         * @see RuntimeException#RuntimeException()
033         */
034        public SimulatedException()
035        {
036        }
037
038        /**
039         * @see RuntimeException#RuntimeException(String)
040         */
041        public SimulatedException(String message)
042        {
043                super(message);
044        }
045}
