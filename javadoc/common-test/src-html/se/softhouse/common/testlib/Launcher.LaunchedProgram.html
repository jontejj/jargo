015package se.softhouse.common.testlib;
016
017import static com.google.common.base.Preconditions.checkArgument;
018import static com.google.common.base.Preconditions.checkNotNull;
019import static java.lang.reflect.Modifier.isPublic;
020import static java.lang.reflect.Modifier.isStatic;
021
022import java.io.File;
023import java.io.IOException;
024import java.util.Arrays;
025import java.util.List;
026import java.util.concurrent.ExecutionException;
027import java.util.concurrent.Future;
028
029import javax.annotation.concurrent.Immutable;
030
031import com.google.common.base.Charsets;
032import com.google.common.collect.Lists;
033
034/**
035 * Can launch java programs for a {@link Class} with a main method. Output is captured in the
036 * returned {@link LaunchedProgram}.
037 */
038@Immutable
039public final class Launcher
040{
041        /**
042         * Result from a {@link Launcher#launch(Class, String...) launched program}
043         */
044        @Immutable
045        public static final class LaunchedProgram
046        {
047                private final String errors;
048                private final String output;
049                private final String debugInformation;
050
051                private LaunchedProgram(String errors, String output, String debugInformation)
052                {
053                        this.errors = errors;
054                        this.output = output;
055                        this.debugInformation = debugInformation;
056                }
057
058                /**
059                 * The {@link System#out} in {@link Charsets#UTF_8 UTF-8} from the launched program
060                 */
061                public String output()
062                {
063                        return output;
064                }
065
066                /**
067                 * The {@link System#err} in {@link Charsets#UTF_8 UTF-8} from the launched program
068                 */
069                public String errors()
070                {
071                        return errors;
072                }
073
074                /**
075                 * Returns information suitable to print in case of errors on a debug level
076                 */
077                public String debugInformation()
078                {
079                        return debugInformation;
080                }
081
082                @Override
083                public String toString()
084                {
085                        return "Output: " + output() + "\nErrors: " + errors();
086                }
087        }
088
089        private Launcher()
090        {
091        }
092
093        /**
094         * Runs {@code classWithMainMethod} in a separate process using the system property
095         * {@code java.home} to find java and {@code java.class.path} for the class path. This method
096         * will wait until program execution has finished.
097         * 
098         * @param classWithMainMethod a class with a static "main" method
099         * @param programArguments optional arguments to pass to the program
100         * @return output/errors from the executed program
101         * @throws IOException if an I/O error occurs while starting {@code classWithMainMethod} as a
102         *             process
103         * @throws InterruptedException if the thread starting the program is
104         *             {@link Thread#interrupted()} while waiting for the program to finish
105         * @throws IllegalArgumentException if {@code classWithMainMethod} doesn't have a public static
106         *             main method
107         * @throws SecurityException if it's not possible to validate the existence of a main method in
108         *             {@code classWithMainMethod} (or if {@link SecurityManager#checkExec checkExec}
109         *             fails)
110         */
111        public static LaunchedProgram launch(Class<?> classWithMainMethod, String ... programArguments) throws IOException, InterruptedException
112        {
113                checkNotNull(programArguments);
114                try
115                {
116
117                        int modifiers = classWithMainMethod.getDeclaredMethod("main", String[].class).getModifiers();
118                        boolean validModifiers = isStatic(modifiers) && isPublic(modifiers);
119                        checkArgument(validModifiers, "%s's main method needs to be static and public for it to be launchable", classWithMainMethod.getName());
120                }
121                catch(NoSuchMethodException e)
122                {
123                        throw new IllegalArgumentException("No main method found on: " + classWithMainMethod.getName(), e);
124                }
125                String jvm = new File(new File(System.getProperty("java.home"), "bin"), "java").getAbsolutePath();
126                String classPath = System.getProperty("java.class.path");
127
128                List<String> args = Lists.newArrayList(jvm, "-cp", classPath, classWithMainMethod.getName());
129                args.addAll(Arrays.asList(programArguments));
130
131                String debugInformation = "\njvm: " + jvm + "\nclasspath: " + classPath;
132                return execute(args, debugInformation);
133        }
134
135        /**
136         * Runs {@code program} with {@code programArguments} as arguments. This method
137         * will wait until program execution has finished.
138         * 
139         * @param program the executable to run
140         * @param programArguments optional arguments to pass to the program
141         * @return output/errors from the executed program
142         * @throws IOException if an I/O error occurs while starting {@code program}
143         * @throws InterruptedException if the thread starting the program is
144         *             {@link Thread#interrupted()} while waiting for the program to finish
145         * @throws SecurityException if {@link SecurityManager#checkExec checkExec} fails
146         */
147        public static LaunchedProgram launch(String program, String ... programArguments) throws IOException, InterruptedException
148        {
149                return execute(Lists.asList(program, programArguments), "Failed to launch " + program);
150        }
151
152        private static LaunchedProgram execute(List<String> args, String debugInformation) throws IOException, InterruptedException
153        {
154                Process program = new ProcessBuilder().command(args).start();
155                Future<String> stdout = Streams.readAsynchronously(program.getInputStream());
156                Future<String> stderr = Streams.readAsynchronously(program.getErrorStream());
157                program.waitFor();
158                try
159                {
160                        String output = stdout.get();
161                        String errors = stderr.get();
162                        return new LaunchedProgram(errors, output, debugInformation);
163                }
164                catch(ExecutionException e)
165                {
166                        if(e.getCause() instanceof IOException)
167                                throw (IOException) e.getCause();
168                        throw new RuntimeException("Failed to read stdout/stderr", e);
169                }
170        }
171}
