035/**
036 * Gives you static access to additional implementations of the {@link Function} interface, as a
037 * complement to {@link Function}
038 */
039@Immutable
040public final class Functions2
041{
042        private Functions2()
043        {
044        }
045
046        /**
047         * Runs several {@link Function}s in the same order as they are
048         * given as arguments here.<br>
049         * If one of the {@link Function}s makes {@code T} {@link Immutable},
050         * make sure to pass it in last as it's hard to modify
051         * an {@link Immutable} value. This works exactly like
052         * {@link Function#compose(Function)} except that if {@code first} doesn't do
053         * anything, {@code second} is returned directly instead.
054         * 
055         * @param first a {@link Function}
056         * @param second another {@link Function}
057         * @return a merged {@link Function}
058         */
059        @Nonnull
060        public static <T> Function<T, T> compound(Function<T, T> first, Function<T, T> second)
061        {
062                requireNonNull(first);
063                requireNonNull(second);
064                if(first == Function.identity())
065                        return second;
066
067                return value -> second.apply(first.apply(value));
068        }
069
070        /**
071         * Creates a {@link Function} that applies {@code elementTransformer} to each element of the
072         * input {@link List} and puts the new elements in a new , immutable, list and returns it.
073         */
074        @Nonnull
075        public static <E> Function<List<E>, List<E>> listTransformer(Function<E, E> elementTransformer)
076        {
077                if(elementTransformer == Function.identity())
078                        return Function.identity();
079                return new ListTransformer<E>(elementTransformer);
080        }
081
082        private static final class ListTransformer<E> implements Function<List<E>, List<E>>
083        {
084                private final Function<E, E> elementTransformer;
085
086                private ListTransformer(Function<E, E> elementTransformer)
087                {
088                        this.elementTransformer = requireNonNull(elementTransformer);
089                }
090
091                @Override
092                public List<E> apply(List<E> values)
093                {
094                        if(values == null)
095                                return null;
096                        return Collections.unmodifiableList(values.stream().map(elementTransformer).collect(toList()));
097                }
098        }
099
100        /**
101         * Creates a {@link Function} that applies {@code valueTransformer} to each value of the
102         * input {@link Map} and puts them in a new , immutable, map and returns it.
103         */
104        @Nonnull
105        public static <K, V> Function<Map<K, V>, Map<K, V>> mapValueTransformer(Function<V, V> valueTransformer)
106        {
107                if(valueTransformer == Function.identity())
108                        return Function.identity();
109                return new MapValueTransformer<K, V>(valueTransformer);
110        }
111
112        private static final class MapValueTransformer<K, V> implements Function<Map<K, V>, Map<K, V>>
113        {
114                private final Function<V, V> valueTransformer;
115
116                private MapValueTransformer(Function<V, V> valueTransformer)
117                {
118                        this.valueTransformer = requireNonNull(valueTransformer);
119                }
120
121                @Override
122                public Map<K, V> apply(Map<K, V> map)
123                {
124                        if(map == null)
125                                return null;
126                        Map<K, V> transformed = new LinkedHashMap<K, V>(map.size());
127                        map.entrySet().stream().forEach((entry) -> transformed.put(entry.getKey(), valueTransformer.apply(entry.getValue())));
128                        return Collections.unmodifiableMap(transformed);
129                }
130        }
131
132        /**
133         * Creates a {@link Function} that wraps {@link List}s with
134         * {@link Collections#unmodifiableList(List)}
135         */
136        @Nonnull
137        public static <E> Function<List<E>, List<E>> unmodifiableList()
138        {
139                return new UnmodifiableListMaker<E>();
140        }
141
142        private static final class UnmodifiableListMaker<E> implements Function<List<E>, List<E>>
143        {
144                @Override
145                public List<E> apply(List<E> value)
146                {
147                        if(value == null)
148                                return null;
149                        return Collections.unmodifiableList(value);
150                }
151        }
152
153        /**
154         * Creates a {@link Function} that wraps {@link Set}s with
155         * {@link Collections#unmodifiableSet(Set)}
156         */
157        @Nonnull
158        public static <E> Function<Set<E>, Set<E>> unmodifiableSet()
159        {
160                return s -> Collections.unmodifiableSet(s);
161        }
162
163        /**
164         * Creates a {@link Function} that wraps {@link Map}s with
165         * {@link Collections#unmodifiableMap(Map)}
166         */
167        @Nonnull
168        public static <K, V> Function<Map<K, V>, Map<K, V>> unmodifiableMap()
169        {
170                return new UnmodifiableMapMaker<K, V>();
171        }
172
173        private static final class UnmodifiableMapMaker<K, V> implements Function<Map<K, V>, Map<K, V>>
174        {
175                @Override
176                public Map<K, V> apply(Map<K, V> value)
177                {
178                        if(value == null)
179                                return null;
180                        return Collections.unmodifiableMap(value);
181                }
182        }
183
184        /**
185         * Runs {@code function} {@code times} times.
186         * The result of the first call is passed to the next apply in the chain and so on.
187         * For instance:
188         * 
189         * <pre class="prettyprint">
190         * <code class="language-java">
191         * assertThat(Functions2.repeat(ADD_ONE, 2).apply(0)).isEqualTo(2);
192         * </code>
193         * </pre>
194         */
195        @Nonnull
196        public static <T> Function<T, T> repeat(Function<T, T> function, long times)
197        {
198                requireNonNull(function);
199                check(times >= 0, "times (%s) must be positive", times);
200                return new FunctionRepeater<T>(function, times);
201        }
202
203        private static final class FunctionRepeater<T> implements Function<T, T>
204        {
205                private final long times;
206                @Nonnull private final Function<T, T> function;
207
208                private FunctionRepeater(Function<T, T> function, long times)
209                {
210                        this.function = function;
211                        this.times = times;
212                }
213
214                @Override
215                public T apply(T input)
216                {
217                        T output = input;
218                        for(long i = 0; i < times; i++)
219                        {
220                                output = function.apply(output);
221                        }
222                        return output;
223                }
224        }
225
226        /**
227         * Returns a {@link Function} that reads whole {@link File}s into {@link String}s using the
228         * {@link StringsUtil#UTF8 UTF-8} charset.
229         */
230        public static Function<File, String> fileToString()
231        {
232                return FileToString.INSTANCE;
233        }
234
235        private static final class FileToString implements Function<File, String>
236        {
237                private static final Function<File, String> INSTANCE = new FileToString();
238
239                @Override
240                public String apply(@Nonnull File input)
241                {
242                        if(input.isDirectory())
243                                throw new IllegalArgumentException(input.getAbsolutePath() + " is a directory, not a file");
244                        try
245                        {
246                                return new String(Files.readAllBytes(input.toPath()), UTF8);
247                        }
248                        catch(IOException e)
249                        {
250                                throw new IllegalArgumentException("I/O error occured while reading: " + input.getAbsolutePath(), e);
251                        }
252                }
253        }
