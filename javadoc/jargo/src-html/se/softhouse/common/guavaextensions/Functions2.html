039/**
040 * Gives you static access to additional implementations of the {@link Function} interface, as a
041 * complement to {@link Functions}
042 */
043@Immutable
044public final class Functions2
045{
046        private Functions2()
047        {
048        }
049
050        /**
051         * Runs several {@link Function}s in the same order as they are
052         * given as arguments here.<br>
053         * If one of the {@link Function}s makes {@code T} {@link Immutable},
054         * make sure to pass it in last as it's hard to modify
055         * an {@link Immutable} value. This works exactly like
056         * {@link Functions#compose(Function, Function)} except that if {@code first} doesn't do
057         * anything, {@code second} is returned directly instead.
058         * 
059         * @param first a {@link Function}
060         * @param second another {@link Function}
061         * @return a merged {@link Function}
062         */
063        @Nonnull
064        public static <T> Function<T, T> compound(Function<T, T> first, Function<T, T> second)
065        {
066                checkNotNull(first);
067                checkNotNull(second);
068                if(first == Functions.identity())
069                        return second;
070
071                return new CompoundFunction<T>(first, second);
072        }
073
074        private static final class CompoundFunction<T> implements Function<T, T>
075        {
076                @Nonnull private final Function<T, T> first;
077                @Nonnull private final Function<T, T> second;
078
079                private CompoundFunction(Function<T, T> first, Function<T, T> second)
080                {
081                        this.first = first;
082                        this.second = second;
083                }
084
085                @Nullable
086                @Override
087                public T apply(@Nullable T value)
088                {
089                        return second.apply(first.apply(value));
090                }
091        }
092
093        /**
094         * Creates a {@link Function} that applies {@code elementTransformer} to each element of the
095         * input {@link List} and puts the new elements in a new , immutable, list and returns it.
096         */
097        @Nonnull
098        public static <E> Function<List<E>, List<E>> listTransformer(Function<E, E> elementTransformer)
099        {
100                if(elementTransformer == Functions.identity())
101                        return Functions.identity();
102                return new ListTransformer<E>(elementTransformer);
103        }
104
105        private static final class ListTransformer<E> implements Function<List<E>, List<E>>
106        {
107                private final Function<E, E> elementTransformer;
108
109                private ListTransformer(Function<E, E> elementTransformer)
110                {
111                        this.elementTransformer = checkNotNull(elementTransformer);
112                }
113
114                @Override
115                public List<E> apply(List<E> values)
116                {
117                        if(values == null)
118                                return null;
119                        return ImmutableList.copyOf(Lists.transform(values, elementTransformer));
120                }
121        }
122
123        /**
124         * Creates a {@link Function} that applies {@code valueTransformer} to each value of the
125         * input {@link Map} and puts them in a new , immutable, map and returns it.
126         */
127        @Nonnull
128        public static <K, V> Function<Map<K, V>, Map<K, V>> mapValueTransformer(Function<V, V> valueTransformer)
129        {
130                if(valueTransformer == Functions.identity())
131                        return Functions.identity();
132                return new MapValueTransformer<K, V>(valueTransformer);
133        }
134
135        private static final class MapValueTransformer<K, V> implements Function<Map<K, V>, Map<K, V>>
136        {
137                private final Function<V, V> valueTransformer;
138
139                private MapValueTransformer(Function<V, V> valueTransformer)
140                {
141                        this.valueTransformer = checkNotNull(valueTransformer);
142                }
143
144                @Override
145                public Map<K, V> apply(Map<K, V> map)
146                {
147                        if(map == null)
148                                return null;
149                        return ImmutableMap.copyOf(Maps.transformValues(map, valueTransformer));
150                }
151        }
152
153        /**
154         * Creates a {@link Function} that wraps {@link List}s with
155         * {@link Collections#unmodifiableList(List)}
156         */
157        @Nonnull
158        public static <E> Function<List<E>, List<E>> unmodifiableList()
159        {
160                return new UnmodifiableListMaker<E>();
161        }
162
163        private static final class UnmodifiableListMaker<E> implements Function<List<E>, List<E>>
164        {
165                @Override
166                public List<E> apply(List<E> value)
167                {
168                        if(value == null)
169                                return null;
170                        return Collections.unmodifiableList(value);
171                }
172        }
173
174        /**
175         * Creates a {@link Function} that wraps {@link Map}s with
176         * {@link Collections#unmodifiableMap(Map)}
177         */
178        @Nonnull
179        public static <K, V> Function<Map<K, V>, Map<K, V>> unmodifiableMap()
180        {
181                return new UnmodifiableMapMaker<K, V>();
182        }
183
184        private static final class UnmodifiableMapMaker<K, V> implements Function<Map<K, V>, Map<K, V>>
185        {
186                @Override
187                public Map<K, V> apply(Map<K, V> value)
188                {
189                        if(value == null)
190                                return null;
191                        return Collections.unmodifiableMap(value);
192                }
193        }
194
195        /**
196         * Runs {@code function} {@code times} times.
197         * The result of the first call is passed to the next apply in the chain and so on.
198         * For instance:
199         * 
200         * <pre class="prettyprint">
201         * <code class="language-java">
202         * assertThat(Functions2.repeat(ADD_ONE, 2).apply(0)).isEqualTo(2);
203         * </code>
204         * </pre>
205         */
206        @Nonnull
207        public static <T> Function<T, T> repeat(Function<T, T> function, long times)
208        {
209                checkNotNull(function);
210                checkArgument(times >= 0, "times (%s) must be positive", times);
211                return new FunctionRepeater<T>(function, times);
212        }
213
214        private static final class FunctionRepeater<T> implements Function<T, T>
215        {
216                private final long times;
217                @Nonnull private final Function<T, T> function;
218
219                private FunctionRepeater(Function<T, T> function, long times)
220                {
221                        this.function = function;
222                        this.times = times;
223                }
224
225                @Override
226                public T apply(T input)
227                {
228                        T output = input;
229                        for(long i = 0; i < times; i++)
230                        {
231                                output = function.apply(output);
232                        }
233                        return output;
234                }
235        }
236
237        /**
238         * Returns a {@link Function} that reads whole {@link File}s into {@link String}s using the
239         * {@link Charsets#UTF_8 UTF-8} charset.
240         */
241        public static Function<File, String> fileToString()
242        {
243                return FileToString.INSTANCE;
244        }
245
246        private static final class FileToString implements Function<File, String>
247        {
248                private static final Function<File, String> INSTANCE = new FileToString();
249
250                @Override
251                public String apply(@Nonnull File input)
252                {
253                        if(input.isDirectory())
254                                throw new IllegalArgumentException(input.getAbsolutePath() + " is a directory, not a file");
255                        try
256                        {
257                                return Files.toString(input, Charsets.UTF_8);
258                        }
259                        catch(IOException e)
260                        {
261                                throw new IllegalArgumentException("I/O error occured while reading: " + input.getAbsolutePath(), e);
262                        }
263                }
264        }
