015package se.softhouse.common.guavaextensions;
016
017import static java.util.Objects.requireNonNull;
018
019import java.util.Collection;
020import java.util.List;
021import java.util.function.Predicate;
022
023import javax.annotation.Nonnull;
024import javax.annotation.concurrent.Immutable;
025
026import se.softhouse.common.strings.Describables;
027
028/**
029 * Additional implementations of the {@link Predicate} interface
030 */
031@Immutable
032public final class Predicates2
033{
034        private Predicates2()
035        {
036        }
037
038        public static <T> Predicate<T> alwaysTrue()
039        {
040                return ObjectPredicates.ALWAYS_TRUE.withNarrowedType();
041        }
042
043        public static <T> Predicate<T> alwaysFalse()
044        {
045                return ObjectPredicates.ALWAYS_FALSE.withNarrowedType();
046        }
047
048        // Visible for testing
049        enum ObjectPredicates implements Predicate<Object>
050        {
051                ALWAYS_TRUE((i) -> true),
052                ALWAYS_FALSE((i) -> false);
053
054                private Predicate<Object> predicate;
055
056                ObjectPredicates(Predicate<Object> predicate)
057                {
058                        this.predicate = predicate;
059                }
060
061                @Override
062                public boolean test(Object o)
063                {
064                        return predicate.test(o);
065                }
066
067                @SuppressWarnings("unchecked") // safe contravariant cast as all ObjectPredicates
068                <T> Predicate<T> withNarrowedType()
069                {
070                        return (Predicate<T>) this;
071                }
072        }
073
074        /**
075         * Returns a predicate that throws an {@link IllegalArgumentException} if any element in a list
076         * doesn't return true for {@code elementLimiter.apply(Object)}. Useful for input validation
077         * where you already have a {@link Predicate} for the element type and want to check a list of
078         * such elements.
079         * Without {@link #listPredicate(Predicate)} it would look something like:
080         *
081         * <pre class="prettyprint">
082         * <code class="language-java">
083         * List&lt;Integer&gt; numbers = Arrays.asList(1, -1, -2);
084         * Optional&lt;Integer&gt; invalidNumber = Iterables.tryFind(numbers, Predicates.not(ABOVE_ZERO));
085         * if(invalidNumber.isPresent())
086         *   throw new IllegalArgumentException("'" + invalidNumber.get() + "' is not " + ABOVE_ZERO);
087         * </code>
088         * </pre>
089         *
090         * With {@link #listPredicate(Predicate)}:
091         *
092         * <pre class="prettyprint">
093         * <code class="language-java">
094         * Predicates2.listPredicate(ABOVE_ZERO).apply(numbers);
095         * </code>
096         * </pre>
097         *
098         * As this breaks the general contract of {@link Predicate#test(Object)} it should be
099         * considered useful for validating arguments only.
100         * The detail message only mentions the first element that didn't return true.
101         */
102        public static <E> Predicate<List<? extends E>> listPredicate(Predicate<E> elementLimiter)
103        {
104                requireNonNull(elementLimiter);
105                if(elementLimiter == alwaysTrue())
106                        return alwaysTrue();
107                return new ListPredicate<E>(elementLimiter);
108        }
109
110        private static final class ListPredicate<E> implements Predicate<List<? extends E>>
111        {
112                private final Predicate<E> elementLimiter;
113
114                private ListPredicate(Predicate<E> elementLimiter)
115                {
116                        this.elementLimiter = elementLimiter;
117                }
118
119                @Override
120                public boolean test(@Nonnull List<? extends E> values)
121                {
122                        for(E value : values)
123                        {
124                                if(!elementLimiter.test(value))
125                                        throw Describables.illegalArgument(Describables.format("'%s' is not %s", value, elementLimiter));
126                        }
127                        return true;
128                }
129
130                @Override
131                public String toString()
132                {
133                        return elementLimiter.toString();
134                }
135        }
136
137        /**
138         * Works just like {@link Predicate#and(Predicate)} except that if {@code first} is
139         * {@link Predicates2#alwaysTrue()} {@code second} is returned directly (or vice versa). This has
140         * the potential to make {@link Object#toString()} look a bit nicer for the resulting
141         * {@link Predicate}.
142         */
143        public static <T> Predicate<? super T> and(Predicate<? super T> first, Predicate<? super T> second)
144        {
145                requireNonNull(first);
146                requireNonNull(second);
147                if(first == alwaysTrue())
148                        return second;
149                else if(second == alwaysTrue())
150                        return first;
151
152                return new Predicate<T>(){
153                        @Override
154                        public boolean test(T o)
155                        {
156                                return first.test(o) && second.test(o);
157                        }
158
159                        @Override
160                        public String toString()
161                        {
162                                return "AND(" + first + ", " + second + ")";
163                        }
164                };
165        }
166
167        /**
168         * Creates a {@link Predicate} which mandates that {@code target} contains the tested object
169         *
170         * @param target the collection that contains the "ok" objects
171         * @param <T> type of objects in {@code target}
172         * @return the newly created {@link Predicate}
173         */
174        public static <T> Predicate<T> in(Collection<? extends T> target)
175        {
176                requireNonNull(target);
177                return (e) -> target.contains(e);
178        }
179}
