015package se.softhouse.common.guavaextensions;
016
017import static com.google.common.base.Preconditions.checkNotNull;
018import static com.google.common.base.Predicates.alwaysTrue;
019
020import java.util.List;
021
022import javax.annotation.Nonnull;
023import javax.annotation.concurrent.Immutable;
024
025import se.softhouse.common.strings.Describables;
026
027import com.google.common.base.Predicate;
028import com.google.common.base.Predicates;
029
030/**
031 * Additional implementations of the {@link Predicate} interface, as a complement to
032 * {@link Predicates}
033 */
034@Immutable
035public final class Predicates2
036{
037        private Predicates2()
038        {
039        }
040
041        /**
042         * Returns a predicate that throws an {@link IllegalArgumentException} if any element in a list
043         * doesn't return true for {@code elementLimiter.apply(Object)}. Useful for input validation
044         * where you already have a {@link Predicate} for the element type and want to check a list of
045         * such elements.
046         * Without {@link #listPredicate(Predicate)} it would look something like:
047         * 
048         * <pre class="prettyprint">
049         * <code class="language-java">
050         * List&lt;Integer&gt; numbers = Arrays.asList(1, -1, -2);
051         * Optional&lt;Integer&gt; invalidNumber = Iterables.tryFind(numbers, Predicates.not(ABOVE_ZERO));
052         * if(invalidNumber.isPresent())
053         *   throw new IllegalArgumentException("'" + invalidNumber.get() + "' is not " + ABOVE_ZERO);
054         * </code>
055         * </pre>
056         * 
057         * With {@link #listPredicate(Predicate)}:
058         * 
059         * <pre class="prettyprint">
060         * <code class="language-java">
061         * Predicates2.listPredicate(ABOVE_ZERO).apply(numbers);
062         * </code>
063         * </pre>
064         * 
065         * As this breaks the general contract of {@link Predicate#apply(Object)} it should be
066         * considered useful for validating arguments only.
067         * The detail message only mentions the first element that didn't return true.
068         */
069        public static <E> Predicate<List<? extends E>> listPredicate(Predicate<E> elementLimiter)
070        {
071                checkNotNull(elementLimiter);
072                if(elementLimiter == alwaysTrue())
073                        return alwaysTrue();
074                return new ListPredicate<E>(elementLimiter);
075        }
076
077        private static final class ListPredicate<E> implements Predicate<List<? extends E>>
078        {
079                private final Predicate<E> elementLimiter;
080
081                private ListPredicate(Predicate<E> elementLimiter)
082                {
083                        this.elementLimiter = elementLimiter;
084                }
085
086                @Override
087                public boolean apply(@Nonnull List<? extends E> values)
088                {
089                        for(E value : values)
090                        {
091                                if(!elementLimiter.apply(value))
092                                        throw Describables.illegalArgument(Describables.format("'%s' is not %s", value, elementLimiter));
093                        }
094                        return true;
095                }
096
097                @Override
098                public String toString()
099                {
100                        return elementLimiter.toString();
101                }
102        }
103
104        /**
105         * Works just like {@link Predicates#and(Predicate, Predicate)} except that if {@code first} is
106         * {@link Predicates#alwaysTrue()} {@code second} is returned directly (or vice versa). This has
107         * the potential to make {@link Predicate#toString()} look a bit nicer for the resulting
108         * {@link Predicate}.
109         */
110        public static <T> Predicate<? super T> and(Predicate<? super T> first, Predicate<? super T> second)
111        {
112                if(first == alwaysTrue())
113                        return checkNotNull(second);
114                else if(second == alwaysTrue())
115                        return checkNotNull(first);
116                return Predicates.<T>and(first, second);
117        }
118}
