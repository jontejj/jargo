015package se.softhouse.common.guavaextensions;
016
017import static java.util.Objects.requireNonNull;
018import static se.softhouse.common.guavaextensions.Preconditions2.check;
019
020import java.io.Serializable;
021import java.util.ArrayList;
022import java.util.List;
023import java.util.function.Supplier;
024
025import javax.annotation.CheckReturnValue;
026import javax.annotation.Nullable;
027import javax.annotation.concurrent.Immutable;
028
029/**
030 * Additional implementations of the {@link Supplier} interface
031 */
032@Immutable
033public final class Suppliers2
034{
035        private Suppliers2()
036        {
037        }
038
039        /**
040         * Creates a {@link Supplier} that supplies {@code elementsToSupply} number of elements from
041         * {@code elementSupplier}
042         *
043         * @throws IllegalArgumentException if {@code elementsToSupply} is less than zero
044         */
045        @CheckReturnValue
046        public static <T> Supplier<List<T>> ofRepeatedElements(Supplier<? extends T> elementSupplier, int elementsToSupply)
047        {
048                requireNonNull(elementSupplier);
049                check(elementsToSupply >= 0, "elementsToSupply may not be negative");
050                return new ListSupplier<T>(elementSupplier, elementsToSupply);
051        }
052
053        private static final class ListSupplier<T> implements Supplier<List<T>>
054        {
055                private final Supplier<? extends T> elementSupplier;
056                private final int elementsToSupply;
057
058                private ListSupplier(Supplier<? extends T> elementSupplier, final int elementsToSupply)
059                {
060                        this.elementSupplier = elementSupplier;
061                        this.elementsToSupply = elementsToSupply;
062                }
063
064                @Override
065                public List<T> get()
066                {
067                        List<T> result = new ArrayList<T>(elementsToSupply);
068                        for(int i = 0; i < elementsToSupply; i++)
069                        {
070                                result.add(elementSupplier.get());
071                        }
072                        return result;
073                }
074        }
075
076        /**
077         * Creates a {@link Supplier} that always returns {@code instance}.
078         * 
079         * @param instance the instance to supply
080         * @param <T> the type
081         * @return the newly created supplier
082         */
083        @CheckReturnValue
084        public static <T> Supplier<T> ofInstance(@Nullable T instance)
085        {
086                return new InstanceSupplier<T>(instance);
087        }
088
089        private static final class InstanceSupplier<T> implements Supplier<T>
090        {
091                private final T instance;
092
093                private InstanceSupplier(T instance)
094                {
095                        this.instance = instance;
096                }
097
098                @Override
099                public T get()
100                {
101                        return instance;
102                }
103        }
104
105        /**
106         * Returns true if {@code Supplier} is likely to supply values very fast
107         */
108        public static boolean isSuppliedAlready(Supplier<?> supplier)
109        {
110                if(supplier.getClass().equals(InstanceSupplier.class))
111                        return true;
112                else if(supplier instanceof ListSupplier<?>)
113                {
114                        ListSupplier<?> listSupplier = (ListSupplier<?>) supplier;
115                        if(listSupplier.elementSupplier.getClass().equals(InstanceSupplier.class))
116                                return true;
117                }
118                return false;
119        }
120
121        /**
122         * Returns a supplier which caches the instance retrieved during the first call to {@code get()}
123         * and returns that value on subsequent calls to {@code get()}. See:
124         * <a href="http://en.wikipedia.org/wiki/Memoization">memoization</a>
125         */
126        public static <T> Supplier<T> memoize(Supplier<T> delegate)
127        {
128                return (delegate instanceof MemoizingSupplier) ? delegate : new MemoizingSupplier<T>(requireNonNull(delegate));
129        }
130
131        private static class MemoizingSupplier<T> implements Supplier<T>, Serializable
132        {
133                final Supplier<T> delegate;
134                transient volatile boolean initialized;
135                // "value" does not need to be volatile; visibility piggy-backs
136                // on volatile read of "initialized".
137                transient T value;
138
139                MemoizingSupplier(Supplier<T> delegate)
140                {
141                        this.delegate = delegate;
142                }
143
144                @Override
145                public T get()
146                {
147                        // A 2-field variant of Double Checked Locking.
148                        if(!initialized)
149                        {
150                                synchronized(this)
151                                {
152                                        if(!initialized)
153                                        {
154                                                T t = delegate.get();
155                                                value = t;
156                                                initialized = true;
157                                                return t;
158                                        }
159                                }
160                        }
161                        return value;
162                }
163
164                private static final long serialVersionUID = 0;
165        }
166}
