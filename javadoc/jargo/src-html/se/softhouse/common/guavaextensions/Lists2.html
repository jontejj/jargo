013package se.softhouse.common.guavaextensions;
014
015import static java.util.Objects.requireNonNull;
016
017import java.util.AbstractList;
018import java.util.ArrayList;
019import java.util.Arrays;
020import java.util.Collection;
021import java.util.Iterator;
022import java.util.List;
023
024import javax.annotation.Nonnull;
025import javax.annotation.Nullable;
026
027public final class Lists2
028{
029        private Lists2()
030        {
031        }
032
033        public static <E> ArrayList<E> newArrayList(Iterable<? extends E> elements)
034        {
035                requireNonNull(elements);
036                return (elements instanceof Collection) ? new ArrayList<E>(cast(elements)) : newArrayList(elements.iterator());
037        }
038
039        public static <E> ArrayList<E> newArrayList(Iterator<? extends E> elements)
040        {
041                ArrayList<E> list = new ArrayList<E>();
042                elements.forEachRemaining((e) -> list.add(e));
043                return list;
044        }
045
046        public static <E> ArrayList<E> asList(@Nonnull E firstElement, E[] rest)
047        {
048                final ArrayList<E> objects = new ArrayList<>();
049                objects.add(requireNonNull(firstElement));
050                objects.addAll(Arrays.asList(rest));
051                return objects;
052        }
053
054        /**
055         * @return {@code true} if the iterable contains no elements
056         */
057        public static boolean isEmpty(Iterable<?> iterable)
058        {
059                if(iterable instanceof Collection)
060                        return ((Collection<?>) iterable).isEmpty();
061                return !iterable.iterator().hasNext();
062        }
063
064        /**
065         * Returns the given {@code CharSequence} as a {@code
066         * List<Character>}
067         *
068         * @param sequence the character sequence to view as a {@code List} of
069         *            characters
070         * @return an {@code List<Character>} view of the character sequence
071         */
072        public static List<Character> charactersOf(CharSequence sequence)
073        {
074                return new CharSequenceAsList(requireNonNull(sequence));
075        }
076
077        private static final class CharSequenceAsList extends AbstractList<Character>
078        {
079                private final CharSequence sequence;
080
081                CharSequenceAsList(CharSequence sequence)
082                {
083                        this.sequence = sequence;
084                }
085
086                @Override
087                public Character get(int index)
088                {
089                        return sequence.charAt(index);
090                }
091
092                @Override
093                public int size()
094                {
095                        return sequence.length();
096                }
097        }
098
099        public static int size(Iterator<?> iterator)
100        {
101                int i = 0;
102                while(iterator.hasNext())
103                {
104                        i++;
105                        iterator.next();
106                }
107                return i;
108        }
109
110        public static int size(Iterable<?> iterable)
111        {
112                if(iterable instanceof Collection)
113                        return ((Collection<?>) iterable).size();
114                return size(iterable.iterator());
115        }
116
117        public static <T> Iterator<T> unmodifiableIterator(final Iterator<? extends T> iterator)
118        {
119                requireNonNull(iterator);
120                return new Iterator<T>(){
121                        @Override
122                        public boolean hasNext()
123                        {
124                                return iterator.hasNext();
125                        }
126
127                        @Override
128                        public T next()
129                        {
130                                return iterator.next();
131                        }
132                };
133        }
134
135        static <T> Collection<T> cast(@Nullable Iterable<T> iterable)
136        {
137                return (Collection<T>) iterable;
138        }
139
140}
