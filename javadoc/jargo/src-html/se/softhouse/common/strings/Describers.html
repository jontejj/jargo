044        /**
045         * Always describes any value of type {@code T} with the given {@code constant}. For instance,
046         * if you have implemented a time related parser and don't want different times in the
047         * usage depending on when the usage is printed you could pass in "The current time", then this
048         * describer would describe any time object with "The current time".
049         */
050        @Nonnull
051        @CheckReturnValue
052        public static <T> Describer<T> withConstantString(final String constant)
053        {
054                requireNonNull(constant);
055                return (value, locale) -> constant;
056        }

058        /**
059         * Calls {@link String#valueOf(Object)} for input values. As this goes
060         * against the very purpose of the {@link Describer} interface it may seem
061         * odd but this {@link Describer} works really well as a null object, it can
062         * also act as a functor for calling toString.
063         */
064        @Nonnull
065        @CheckReturnValue
066        public static <T> Describer<T> toStringDescriber()
067        {
068                return (value, locale) -> String.valueOf(value);
069        }

082        {
083                private static final Describer<Character> INSTANCE = new CharDescriber();

085                @Override
086                public String describe(Character value, Locale inLocale)

088                        if(value == null)

188        /**
189         * Describes what a key=value in a {@link Map} means by fetching a
190         * description from {@code descriptions} for each key in a given map. So {@code descriptions}
191         * need to have values (descriptions) for all keys in any given map,
192         * otherwise a {@link NullPointerException} is thrown. For example:
193         * 
194         * <pre class="prettyprint">
195         * <code class="language-java">
196         * Map&lt;String, Integer&gt; defaults = newLinkedHashMap();
197         * defaults.put("population", 42);
198         * defaults.put("hello", 1);
199         * 
200         * Map&lt;String, String&gt; descriptions = newLinkedHashMap();
201         * descriptions.put("population", "The number of citizens in the world");
202         * descriptions.put("hello", "The number of times to say hello");
203         * Describer&lt;Map&lt;String, Integer&gt;&gt; d = mapDescriber(descriptions);
204         * 
205         * String describedMap = d.describe(defaults);
206         * </code>
207         * </pre>
208         * 
209         * would return:
210         * 
211         * <pre>
212         * <code>
213         * population=42
214         *      The number of citizens in the world
215         * hello=1
216         *      The number of times to say hello
217         * </code>
218         * </pre>
219         * 
220         * You could even implement a
221         * {@code addProperty(String key, int defaultValue, String description)} method to enforce the
222         * use of descriptions at compile-time.
223         */
224        @Nonnull
225        @CheckReturnValue
226        public static <K, V> Describer<Map<K, V>> mapDescriber(Map<K, String> descriptions)
227        {
228                return new MapDescription<K, V>(unmodifiableMap(descriptions), Describers.<K>toStringDescriber());
229        }
230
231        /**
232         * Works like {@link #mapDescriber(Map)} but it describes keys in any given {@link Map} with
233         * {@code keyDescriber} instead of with {@link #toString()}
234         * 
235         * @param descriptions a map with strings describing what each key in it means
236         * @param keyDescriber {@link Describer} used to transform keys into {@link String}s with
237         * @return a {@link Describer} that can describe a map of the type {@code Map<K, V>} and where
238         *         each key must have a corresponding description in {@code descriptions}
239         */
240        public static <K, V> Describer<Map<K, V>> mapDescriber(Map<K, String> descriptions, Describer<K> keyDescriber)
241        {
242                return new MapDescription<K, V>(unmodifiableMap(descriptions), requireNonNull(keyDescriber));
243        }
244
245        private static final class MapDescription<K, V> implements Describer<Map<K, V>>
246        {
247                private final Map<K, String> descriptions;
248                private final Describer<K> keyDescriber;
249
250                private MapDescription(Map<K, String> descriptions, Describer<K> keyDescriber)
251                {
252                        this.descriptions = descriptions;
253                        this.keyDescriber = keyDescriber;
254                }
255
256                @Override
257                public String describe(Map<K, V> values, Locale inLocale)
258                {
259                        StringBuilder result = new StringBuilder();
260                        for(Entry<K, V> entry : values.entrySet())
261                        {
262                                K key = entry.getKey();
263                                result.append(keyDescriber.describe(key, inLocale));
264                                result.append("=");
265                                result.append(entry.getValue());
266                                String descriptionForEntry = descriptions.get(key);
267                                check(descriptionForEntry != null, "Undescribed key: %s", key);
268                                result.append(NEWLINE).append(" ").append(descriptionForEntry).append(NEWLINE);
269                        }
270                        return result.toString();
271                }
272        }

316         */
317        @CheckReturnValue
318        @Nonnull
319        public static <K, V> Describer<Map<K, V>> mapDescriber(Describer<K> keyDescriber, Describer<V> valueDescriber, String valueSeparator)
320        {
321                return new MapDescriber<K, V>(keyDescriber, valueDescriber, valueSeparator);
322        }

324        private static final class MapDescriber<K, V> implements Describer<Map<K, V>>
325        {
326                private final Describer<V> valueDescriber;
327                private final Describer<K> keyDescriber;
328                private final String valueSeparator;
329
330                private MapDescriber(Describer<K> keyDescriber, Describer<V> valueDescriber, String valueSeparator)
331                {
332                        this.keyDescriber = requireNonNull(keyDescriber);
333                        this.valueDescriber = requireNonNull(valueDescriber);
334                        this.valueSeparator = requireNonNull(valueSeparator);
335                }
336
337                @Override
338                public String describe(Map<K, V> values, Locale inLocale)
339                {
340                        if(values == null)
341                                return NULL;
342                        Iterator<Entry<K, V>> iterator = values.entrySet().iterator();
343                        if(!iterator.hasNext())
344                                return "Empty map";
345
346                        StringBuilder firstKeyValue = new StringBuilder();
347                        describeEntry(iterator.next(), inLocale, firstKeyValue);
348
349                        StringBuilder result = StringBuilders.withExpectedSize(firstKeyValue.length() * values.size());
350                        result.append(firstKeyValue);
351
352                        while(iterator.hasNext())
353                        {
354                                result.append(NEWLINE);
355                                describeEntry(iterator.next(), inLocale, result);
356                        }
357                        return result.toString();
358                }
359
360                private void describeEntry(Entry<K, V> entry, Locale inLocale, StringBuilder output)
361                {
362                        output.append(keyDescriber.describe(entry.getKey(), inLocale));
363                        output.append(valueSeparator);
364                        output.append(valueDescriber.describe(entry.getValue(), inLocale));
365                }
366        }

368        /**
369         * Like {@link Describer#apply(Object)} but with {@code locale} instead of
370         * {@link Locale#getDefault()}.
371         */
372        @Nonnull
373        @CheckReturnValue
374        public static <T> Function<? super T, String> asFunction(final Describer<T> describer, final Locale locale)
375        {
376                requireNonNull(describer);
377                requireNonNull(locale);
378                return (Function<T, String>) input -> describer.describe(input, locale);
379        }

381        /**
382         * <pre>
383         * Exposes a {@link Function} as a {@link Describer}.
384         *
385         * &#64;param describerFunction a function that can convert {@code T} values into {@link String}s
386         * @return a {@link Describer} that applies {@link Function#apply(Object)} to {@link Describer#describe(Object, Locale)} input values.
387         * </pre>
388         */
389        @Nonnull
390        @CheckReturnValue
391        public static <T> Describer<T> usingFunction(final Function<T, String> describerFunction)
392        {
393                requireNonNull(describerFunction);
394                return (value, inLocale) -> describerFunction.apply(value);
395        }

409        /**
410         * Describes values in lists with {@code valueDescriber} and separates

412         * is described with the string "Empty List".
413         */
414        @Nonnull
415        @CheckReturnValue
416        public static <T> Describer<List<? extends T>> listDescriber(Describer<T> valueDescriber, String valueSeparator)

418                return new ListDescriber<T>(valueDescriber, valueSeparator);
419        }
420
421        private static final class ListDescriber<T> implements Describer<List<? extends T>>
422        {
423                private final Describer<T> valueDescriber;
424                private final String valueSeparator;
425
426                ListDescriber(Describer<T> valueDescriber, String valueSeparator)
427                {
428                        this.valueDescriber = requireNonNull(valueDescriber);
429                        this.valueSeparator = requireNonNull(valueSeparator);
430                }
431
432                @Override
433                public String describe(List<? extends T> value, Locale inLocale)
434                {
435                        if(value == null)
436                                return NULL;
437                        if(value.isEmpty())
438                                return "Empty list";
439
440                        Iterator<? extends T> values = value.iterator();
441                        String firstValue = valueDescriber.describe(values.next(), inLocale);
442
443                        StringBuilder sb = StringBuilders.withExpectedSize(value.size() * firstValue.length());
444
445                        sb.append(firstValue);
446                        while(values.hasNext())
447                        {
448                                sb.append(valueSeparator).append(valueDescriber.describe(values.next(), inLocale));
449                        }
450                        return sb.toString();
451                }
452        }
