069/**
070 * <pre>
071 * Responsible for configuring and building {@link Argument} instances.
072 * Example builders can be created with {@link Arguments}.
073 *
074 * <b>Note:</b>The code examples (for each method) assumes that all methods in {@link Arguments} have been statically imported.
075 *
076 * <b>Note:</b>Some methods needs to be called in a specific order
077 * (to make the generic type system produce the correct type) and to guide the
078 * caller, such invalid orders are documented with {@link Deprecated}. If those warnings
079 * are ignored {@link IllegalStateException} will be thrown at the offending call.
080 * </pre>
081 *
082 * @param <SELF> the type of the subclass extending this class.
083 *            Concept borrowed from: <a href=
084 *            "http://passion.forco.de/content/emulating-self-types-using-java-generics-simplify-fluent-api-implementation">Ansgar.Konermann's
085 *            blog</a> The pattern also resembles the
086 *            <a href="http://en.wikipedia.org/wiki/Curiously_recurring_template_pattern">Curiously
087 *            recurring template pattern</a>
088 * @param <T> the type of arguments the built {@link Argument} instance should handle,
089 *            such as {@link Integer} in the case of {@link Arguments#integerArgument(String...)}
090 */
091@NotThreadSafe
092public abstract class ArgumentBuilder<SELF extends ArgumentBuilder<SELF, T>, T>
093{
094        static final String DEFAULT_SEPARATOR = " ";

100        private boolean ignoreCase = false;
101        private boolean isAllowedToRepeat = false;
102        @Nonnull private Optional<String> metaDescription = Optional.empty();
103        private boolean hideFromUsage = false;
104        @Nullable private Function<String, Set<String>> completer;

112        @Nonnull private Function<T, T> finalizer = Function.identity();

119        /**
120         * Creates a default {@link ArgumentBuilder} that produces arguments that:
121         * <ul>
122         * <li>doesn't have any {@link #names(String...)}</li>
123         * <li>has an empty {@link #description(String)}</li>
124         * <li>isn't {@link #required()}</li>
125         * <li>uses {@link StringParser#defaultValue()} on {@link #parser()} to produce default
126         * values</li>
127         * <li>uses {@link Object#toString()} to describe the default value</li>
128         * <li>uses {@link StringParser#metaDescription()} on {@link #parser()} to produce meta
129         * descriptions</li>
130         * <li>doesn't have any {@link #limitTo(Predicate) limits}</li>
131         * </ul>
132         * Typically invoked implicitly by subclasses.
133         */
134        protected ArgumentBuilder()
135        {
136                this(null);
137        }

139        ArgumentBuilder(@Nullable final InternalStringParser<T> stringParser)
140        {
141                this.internalStringParser = stringParser;
142                myself = self();
143        }

169        /**
170         * <pre>
171         * Parses command line arguments and returns the value of the argument built
172         * by {@link #build()}, providing a simple one liner shortcut when faced with only one argument.
173         * </pre>
174         *
175         * <pre class="prettyprint">
176         * <code class="language-java">
177         * String[] args = {"--listen-port", "8090"};
178         *
179         * int port = integerArgument("-p", "--listen-port").defaultValue(8080).description("The port clients should connect to.").parse(args);
180         * assertThat(port).isEqualTo(8090);
181         * </code>
182         * </pre>
183         *
184         * This is a shorthand method that should be used if only one {@link Argument} is expected as it
185         * will throw if unexpected arguments are encountered. If several arguments are expected use
186         * {@link CommandLineParser#withArguments(Argument...)} instead.
187         *
188         * @param actualArguments the arguments from the command line
189         * @return the parsed value from the {@code actualArguments}
190         * @throws ArgumentException if actualArguments isn't compatible with this
191         *             argument
192         */
193        @Nullable
194        public final T parse(String ... actualArguments) throws ArgumentException
195        {
196                return build().parse(actualArguments);
197        }

200         * Returns a usage string for the argument that this builder builds.
201         * Should only be used if one argument is supported,
202         * otherwise the {@link CommandLineParser#usage()} method should be used instead.
203         */
204        public final Usage usage()
205        {
206                return build().usage();
207        }

209        /**
210         * <pre>
211         * Returns a customized parser that the {@link Argument} will use to parse values.
212         *
213         * This is a suitable place to verify the configuration of your parser.
214         *
215         * If your {@link StringParser} doesn't support any configuration you can use
216         * {@link Arguments#withParser(StringParser)} directly instead of subclassing
217         * {@link ArgumentBuilder}
218         *
219         * &#64;return the {@link StringParser} that performs the actual parsing of an argument value
220         * &#64;throws IllegalStateException if the parser have been configured wrongly
221         * </pre>
222         */

224        protected abstract StringParser<T> parser();
225
226        @Nonnull
227        InternalStringParser<T> internalParser()
228        {
229                StringParser<T> parser = parser();
230                if(parser != InternalArgumentBuilder.MARKER)
231                        return new StringParserBridge<T>(parser);
232                return internalStringParser;
233        }

314         * <pre>
315         * Makes {@link CommandLineParser#parse(String...)} throw
316         * {@link ArgumentException} if this argument isn't given.
317         * If possible, it's preferred to use a {@link #defaultValue(Object) default value} instead.
318         *
319         * The {@link Argument#toString()} will be used to print each missing argument.
320         *
321         * <b>Note</b>: If you choose to use multiple {@link #required()} indexed arguments all of them
322         * must have unique {@link #metaDescription(String)}s. This ensures that error messages
323         * can point out erroneous arguments better
324         *
325         * &#64;return this builder
326         * @throws IllegalStateException if {@link #defaultValue(Object)} (or
327         *             {@link #defaultValueSupplier(Supplier)}) has been
328         *             called, because these are mutually exclusive with {@link #required()}
329         * </pre>

338        /**
339         * <pre>
340         * Sets a default value to use for this argument. Overrides {@link StringParser#defaultValue()} which is used by default.
341         * Returned by {@link ParsedArguments#get(Argument)} when no argument {@link ParsedArguments#wasGiven(Argument) was given}.
342         * To create default values lazily see {@link ArgumentBuilder#defaultValueSupplier(Supplier)}.
343         * </pre>
344         *
345         * <b>Mutability</b>:Remember that as {@link Argument} is {@link Immutable}
346         * this value should be so too if multiple argument parsings is going to take place.
347         * If mutability is wanted {@link ArgumentBuilder#defaultValueSupplier(Supplier)} should be used
348         * instead.
349         *
350         * @return this builder
351         * @throws IllegalStateException if {@link #required()} has been called,
352         *             because these two methods are mutually exclusive
353         */
354        public SELF defaultValue(@Nullable final T value)
355        {
356                check(!required, ProgrammaticErrors.DEFAULT_VALUE_AND_REQUIRED);
357                defaultValueSupplier = Suppliers2.ofInstance(value);
358                return myself;
359        }

367         * called
368         * until the default value is actually needed ({@link ParsedArguments#get(Argument)}. If the
369         * default value is deemed non-allowed at that point an {@link IllegalStateException} is thrown.
370         * Wrap your supplier with {@link Suppliers2#memoize(Supplier)} if you want to cache created

372         *

375         *             because these two methods are mutually exclusive
376         */

379                check(!required, ProgrammaticErrors.DEFAULT_VALUE_AND_REQUIRED);

411        /**
412         * <pre>
413         * By default {@link StringParser}s provides a meta description (by implementing {@link StringParser#metaDescription()}
414         * that describes the type of data they expect. For instance, if you're writing a music player,
415         * a user of your application would (most probably) rather see:
416         *
417         * --track-nr &lt;track nr&gt;    The track number to play (using {@code  metaDescription("<track nr>") })
418         *
419         * instead of
420         *
421         * --track-nr &lt;integer&gt;     The track number to play (the default provided by {@link StringParser#metaDescription()})
422         *
423         * So when using general data type parsers such as {@link StringParsers#integerParser()} you're better of if
424         * you provide a meta description that explains what the {@code integer} represents in the context of this argument.
425         *
426         * <b>Note:</b> empty meta descriptions aren't allowed
427         * <b>Note:</b> the surrounding &lt; and &gt; aren't enforced or added automatically but it's preferred
428         * to have them because it makes a clear distinction between {@link #names(String...) argument names} and their parameters.
429         *
430         * &#64;param aMetaDescription "&lt;track nr&gt;" in the above example
431         * &#64;return this builder
432         * &#64;throws IllegalArgumentException if aMetaDescription is empty
433         * </pre>
434         */
435        public final SELF metaDescription(final String aMetaDescription)
436        {
437                check(aMetaDescription.length() > 0, ProgrammaticErrors.INVALID_META_DESCRIPTION);
438                this.metaDescription = Optional.of(aMetaDescription);
439                return myself;
440        }

469        /**
470         * <pre>
471         * Limits values parsed so that they conform to some specific rule.
472         * Only values for which {@link Predicate#test(Object)} returns true, will be accepted.
473         * Other values will cause an {@link ArgumentException}.
474         *
475         * To override the default error message that is generated with {@link UserErrors#DISALLOWED_VALUE}
476         * you can throw {@link IllegalArgumentException} from {@link Predicate#test(Object)}. The detail
477         * message of that exception will be used by {@link ArgumentException#getMessageAndUsage()}.
478         * When this is needed it's generally recommended to write a parser of its own instead.
479         *
480         * <b>Descriptions:</b>{@link Object#toString() toString()} on {@code aLimiter} will replace {@link StringParser#descriptionOfValidValues(Locale)} in the usage. 
481         * A simpler alternative to provide a description is to use {@link #limitTo(Predicate, String)}.
482         *
483         * <b>Supplied default values:</b>The validity of any {@link #defaultValueSupplier(Supplier) default value} isn't checked until
484         * it's actually needed when {@link ParsedArguments#get(Argument)} is called. This is so
485         * because {@link Supplier#get()} (or {@link StringParser#defaultValue()}) could take an arbitrary long time.
486         *
487         * <b>Many limiters:</b>Any previously set limiter will be {@link Predicates2#and(Predicate, Predicate)
488         * and'ed} together with {@code aLimiter}.
489         * 
490         * <b>Interopability with completers</b> Values suggested with any {@link #completer(Function)} will not be limited automatically. 
491         * So before generating the suggestions, you should make sure that they are confirming to your limiter.
492         * </pre>
493         *
494         * @param aLimiter a limiter
495         * @return this builder
496         */
497        public SELF limitTo(Predicate<? super T> aLimiter)
498        {
499                limiter = Predicates2.and(limiter, aLimiter);
500                return myself;
501        }
502
503        /**
504         * Like {@link #limitTo(Predicate)} but with {@code aDescription} to describe the valid values this limiter accepts
505         * 
506         * @return this builder
507         */
508        public SELF limitTo(Predicate<? super T> aLimiter, String aDescription)
509        {
510                return limitTo(aLimiter, Describables.withString(aDescription));
511        }
512
513        /**
514         * Like {@link #limitTo(Predicate, String)} but allows for lazy construction of the description
515         */
516        public SELF limitTo(Predicate<? super T> aLimiter, Describable aDescription)
517        {
518                requireNonNull(aLimiter);
519                requireNonNull(aDescription);
520                Predicate<? super T> withDescription = new Predicate<T>(){
521
522                        @Override
523                        public boolean test(T t)
524                        {
525                                return aLimiter.test(t);
526                        }
527
528                        @Override
529                        public String toString()
530                        {
531                                return aDescription.description();
532                        }
533                };
534                limitTo(withDescription);
535                return myself;
536        }

562        /**
563         * <pre>
564         * Makes this argument handle "property like" arguments:
565         * -Dproperty_name=value
566         * where
567         * "-D" is the string supplied to {@link #names(String...)},
568         * "property_name" is the {@link String} key in the resulting {@link Map},
569         * "=" is the {@link #separator(String)} (set to "=" if it hasn't been overridden already),
570         * "value" is decoded by the previously set {@link StringParser}.
571         *
572         *
573         * Tip: You can pass {@link #defaultValueDescriber(Describer)} a
574         * {@link Describers#mapDescriber(Map)} to describe each property that you
575         * support (given that you've used {@link ArgumentBuilder#defaultValue(Object)}
576         * to use sane defaults for your properties).
577         *
578         * &#64;return a new (more specific) builder
579         *
580         * @see #asKeyValuesWithKeyParser(StringParser) to use other types as keys than {@link String}
581         * </pre>
582         */
583        @CheckReturnValue
584        public final MapArgumentBuilder<String, T> asPropertyMap()
585        {
586                return new MapArgumentBuilder<String, T>(this, stringParser());
587        }

589        /**
590         * <pre>
591         * Makes this argument handle properties like arguments:
592         * -Dproperty_name=value
593         * where "-D" is one of the strings supplied to {@link #names(String...)},
594         * "property_name" is decoded by {@code keyParser} and
595         * "value" is decoded by the {@link StringParser} previously passed to the constructor.
596         *
597         * For example:
598         * </pre>
599         *
600         * <pre class="prettyprint">
601         * <code class="language-java">
602         * Map&lt;Integer, Integer&gt; numberMap = Arguments.integerArgument("-N")
603         *                                              .asKeyValuesWithKeyParser(StringParsers.integerParser())
604         *                                              .parse("-N1=5", "-N2=10");
605         * assertThat(numberMap.get(1)).isEqualTo(5);
606         * </code>
607         * </pre>
608         *
609         * For this to work correctly it's paramount that {@code K} implements a
610         * proper {@link Object#hashCode()} because it's going to be used a key in a {@link Map}.
611         *
612         * @return a new (more specific) builder
613         */
614        @CheckReturnValue
615        public final <K> MapArgumentBuilder<K, T> asKeyValuesWithKeyParser(StringParser<K> keyParser)
616        {
617                return new MapArgumentBuilder<K, T>(this, requireNonNull(keyParser));
618        }

621         * <pre>
622         * When given a "," this allows for
623         * arguments such as:
624         * -numbers 1,2,3
625         * where the resulting {@code List&lt;Integer&gt;} would contain 1, 2 and 3.
626         *
627         * <b>Note:</b> Doesn't allow empty lists. Leading and trailing whitespace will be trimmed for each element.
628         * </pre>
629         *
630         * @param valueSeparator the string to split the input with

1268        {

1270
1271                private final ArgumentBuilder<?, V> valueBuilder;
1272                private final StringParser<K> keyParser;

1275                {

1277                        this.keyParser = keyParser;

1288                {
1289                        check(names().size() > 0, ProgrammaticErrors.NO_NAME_FOR_PROPERTY_MAP);
1290                        if(separator().equals(DEFAULT_SEPARATOR))
1291                        {
1292                                separator(DEFAULT_KV_SEPARATOR);

1294                        else
1295                        {
1296                                check(separator().length() > 0, ProgrammaticErrors.EMPTY_SEPARATOR);
1297                        }
1298                        return new KeyValueParser<K, V>(keyParser,
1299                                        valueBuilder.internalParser(),
1300                                        valueBuilder.limiter,

1309                {
1310                        Describer<? super Map<K, V>> overriddenDescriber = super.defaultValueDescriber();
1311                        if(overriddenDescriber != null)
1312                                return overriddenDescriber;
1313
1314                        Describer<? super V> valueDescriber = valueBuilder.defaultValueDescriber();
1315                        if(valueDescriber != null)
1316                        {
1317                                Describer<?> mapDescriber = Describers.mapDescriber(valueDescriber, separator());

1410         * An intermediate builder used by {@link #transform(Function)}. It's used to switch the
1411         * {@code <T>}
1412         * argument of the previous builder to {@code <F>} and to indicate invalid call orders.
1413         * 
1414         * @param <F> The new type
1415         */
1416        @NotThreadSafe
1417        public static final class TransformingArgumentBuilder<F> extends InternalArgumentBuilder<TransformingArgumentBuilder<F>, F>
1418        {
1419                private <T> TransformingArgumentBuilder(final ArgumentBuilder<?, T> builder, final Function<T, F> transformer)
1420                {
1421                        super(new TransformingParser<T, F>(builder.internalParser(), transformer, builder.limiter()));
1422                        copy(builder);
1423
1424                        Supplier<? extends T> defaultValueSupplier = builder.defaultValueSupplierOrFromParser();
1425                        defaultValueSupplier(Suppliers2.wrapWithPredicateAndTransform(defaultValueSupplier, transformer, builder.limiter()));
1426
1427                        if(builder.defaultValueDescriber() != null)
1428                        {
1429                                defaultValueDescriber(new BeforeTransformationDescriber<>(defaultValueSupplier, builder.defaultValueDescriber()));
1430                        }
1431                }
1432        }

1434        private static final class BeforeTransformationDescriber<F> implements Describer<Object>
1435        {
1436                private final Supplier<? extends F> valueProvider;
1437                private final Describer<F> beforeDescriber;
1438
1439                BeforeTransformationDescriber(Supplier<? extends F> valueProvider, Describer<F> beforeDescriber)
1440                {
1441                        this.valueProvider = requireNonNull(valueProvider);
1442                        this.beforeDescriber = requireNonNull(beforeDescriber);
1443                }
1444
1445                @Override
1446                public String describe(Object value, Locale inLocale)
1447                {
1448                        F beforeValue = valueProvider.get();
1449                        return beforeDescriber.apply(beforeValue);
1450                }
