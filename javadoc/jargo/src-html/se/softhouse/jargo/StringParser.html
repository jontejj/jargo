015package se.softhouse.jargo;
016
017import java.util.Locale;
018import java.util.function.Function;
019
020import javax.annotation.Nonnull;
021import javax.annotation.Nullable;
022import javax.annotation.concurrent.Immutable;
023
024/**
025 * Parses {@link String}s into values of the type {@code T}.
026 * Create an {@link Argument} for your {@link StringParser} with
027 * {@link Arguments#withParser(StringParser)}.
028 * An example implementation for handling joda-time dates:
029 *
030 * <pre class="prettyprint">
031 * <code class="language-java">
032 * public class DateTimeParser implements StringParser&lt;DateTime&gt;
033 * {
034 *      public static DefaultArgumentBuilder&lt;DateTime&gt; dateArgument(String ... names)
035 *      {
036 *              return Arguments.withParser(new DateTimeParser()).defaultValueDescription("Current time").names(names);
037 *      }
038 *
039 *      public String descriptionOfValidValues()
040 *      {
041 *              return "an ISO8601 date, such as 2011-02-28";
042 *      }
043 *
044 *      public DateTime parse(String value, Locale locale) throws ArgumentException
045 *      {
046 *              try
047 *              {
048 *                      return DateTime.parse(value);
049 *              }
050 *              catch(IllegalArgumentException wrongDateFormat)
051 *              {
052 *                      throw ArgumentExceptions.withMessage(wrongDateFormat.getLocalizedMessage());
053 *              }
054 *      }
055 *
056 *      public DateTime defaultValue()
057 *      {
058 *              return DateTime.now();
059 *      }
060 *
061 *      public String metaDescription()
062 *      {
063 *              return "&lt;date&gt;";
064 *      }
065 * }
066 * </code>
067 * </pre>
068 *
069 * <pre>
070 * When printed in usage, this would look like:
071 * <code>
072 * --start &lt;date&gt; &lt;date&gt;: an ISO8601 date, such as 2011-02-28
073 *                Default: Current time
074 * </code>
075 * </pre>
076 *
077 * Btw, this parser can be found in jargo-addons.
078 *
079 * @param <T> the type to parse {@link String}s into
080 * @see StringParsers
081 */
082@Immutable
083public interface StringParser<T> extends Function<String, T>
084{
085        /**
086         * Parses the given {@link String} into the type {@code T}
087         *
088         * @param argument the string as given from the command line
089         * @param locale the locale to parse strings with, may matter when parsing numbers, dates etc
090         * @return the parsed value
091         * @throws ArgumentException if the string isn't valid according to
092         *             {@link #descriptionOfValidValues(Locale)}
093         */
094        @Nonnull
095        T parse(String argument, Locale locale) throws ArgumentException;
096
097        /**
098         * <pre>
099         * Uses {@link StringParser#parse(String, Locale) parse} for input {@link String}s.
100         * </pre>
101         *
102         * For example:
103         *
104         * <pre class="prettyprint">
105         * <code class="language-java">
106         * List&lt;Integer&gt; result =  Arrays.asList("1", "3", "2").stream().map(StringParsers.integerParser()).collect(Collectors.toList());
107         * </code>
108         * </pre>
109         *
110         * <b>Note:</b>The parser will pass the {@link Locale#getDefault()} to
111         * {@link StringParser#parse(String, Locale) parse}.
112         * 
113         * @param argument the string to convert
114         * @return the parsed value
115         */
116        @Override
117        default T apply(String argument)
118        {
119                return parse(argument, Locale.getDefault());
120        }
121
122        /**
123         * Describes what values this {@link StringParser} accepts
124         *
125         * @param locale the locale to print the description with
126         * @return a description string to show in usage texts
127         */
128        @Nonnull
129        String descriptionOfValidValues(Locale locale);
130
131        /**
132         * Used when the {@link Argument} this parser is connected to wasn't given.
133         * Used as a fall-back mechanism when {@link ArgumentBuilder#defaultValue(Object)} isn't used.
134         */
135        @Nullable
136        T defaultValue();
137
138        /**
139         * <pre>
140         * The meta description is the text displayed after the argument names.
141         * Typically surrounded by &lt; and &gt;
142         * Sort of like the default value of {@link ArgumentBuilder#metaDescription(String)}.
143         *
144         * For instance:
145         * {@link StringParsers#integerParser()}s {@link #metaDescription()} is &lt;integer&gt;
146         * </pre>
147         *
148         * @return a meta description that very briefly explains what value this parser expects
149         */
150        @Nonnull
151        String metaDescription();
152}
