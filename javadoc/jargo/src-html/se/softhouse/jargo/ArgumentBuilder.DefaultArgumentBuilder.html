069/**
070 * <pre>
071 * Responsible for configuring and building {@link Argument} instances.
072 * Example builders can be created with {@link Arguments}.
073 * 
074 * <b>Note:</b>The code examples (for each method) assumes that all methods in {@link Arguments} have been statically imported.
075 * 
076 * <b>Note:</b>Some methods needs to be called in a specific order
077 * (to make the generic type system produce the correct type) and to guide the
078 * caller, such invalid orders are documented with {@link Deprecated}. If those warnings
079 * are ignored {@link IllegalStateException} will be thrown at the offending call.
080 * 
081 * @param <SELF> the type of the subclass extending this class.
082 *              Concept borrowed from: <a href="http://passion.forco.de/content/emulating-self-types-using-java-generics-simplify-fluent-api-implementation">Ansgar.Konermann's blog</a>
083 *              The pattern also resembles the <a href="http://en.wikipedia.org/wiki/Curiously_recurring_template_pattern">Curiously recurring template pattern</a>
084 * @param <T> the type of arguments the built {@link Argument} instance should handle,
085 *      such as {@link Integer} in the case of {@link Arguments#integerArgument(String...)}
086 * </pre>
087 */
088@NotThreadSafe
089public abstract class ArgumentBuilder<SELF extends ArgumentBuilder<SELF, T>, T>
090{
091        static final String DEFAULT_SEPARATOR = " ";
092        // ArgumentSetting variables, think about #copy() when adding new ones
093        @Nonnull private List<String> names = emptyList();
094        @Nonnull private Description description = EMPTY_STRING;
095        private boolean required = false;
096        @Nonnull private String separator = DEFAULT_SEPARATOR;
097        @Nonnull private Optional<Locale> localeOverride = Optional.absent();
098        private boolean ignoreCase = false;
099        private boolean isAllowedToRepeat = false;
100        @Nonnull private Optional<String> metaDescription = Optional.absent();
101        private boolean hideFromUsage = false;
102
103        private boolean isPropertyMap = false;
104        private ParameterArity parameterArity = ParameterArity.AT_LEAST_ONE_ARGUMENT;

107        // ListArgumentBuilder#copyAsListBuilder() when adding new ones
108        @Nullable private Supplier<? extends T> defaultValueSupplier = null;
109        @Nullable private Describer<? super T> defaultValueDescriber = null;
110        @Nonnull private Function<T, T> finalizer = Functions.identity();
111        @Nonnull private Predicate<? super T> limiter = Predicates.alwaysTrue();

117        /**
118         * Creates a default {@link ArgumentBuilder} that produces arguments that:
119         * <ul>
120         * <li>doesn't have any {@link #names(String...)}</li>
121         * <li>has an empty {@link #description(String)}</li>
122         * <li>isn't {@link #required()}</li>
123         * <li>uses {@link StringParser#defaultValue()} on {@link #parser()} to produce default values</li>
124         * <li>uses {@link Object#toString()} to describe the default value</li>
125         * <li>uses {@link StringParser#metaDescription()} on {@link #parser()} to produce meta
126         * descriptions</li>
127         * <li>doesn't have any {@link #limitTo(Predicate) limits}</li>
128         * </ul>
129         * Typically invoked implicitly by subclasses.
130         */
131        protected ArgumentBuilder()
132        {
133                this(null);
134        }
135
136        ArgumentBuilder(final InternalStringParser<T> stringParser)
137        {
138                this.internalStringParser = stringParser;
139                myself = self();
140        }

154         * argument can be fetched with {@link ParsedArguments#get(Argument)}.

157         *             current settings

163                return new Argument<T>(this);
164        }

166        /**
167         * <pre>
168         * Parses command line arguments and returns the value of the argument built
169         * by {@link #build()}, providing a simple one liner shortcut when faced with only one argument.
170         * </pre>
171         * 
172         * <pre class="prettyprint">
173         * <code class="language-java">
174         * String[] args = {"--listen-port", "8090"};
175         * 
176         * int port = integerArgument("-p", "--listen-port").defaultValue(8080).description("The port clients should connect to.").parse(args);
177         * assertThat(port).isEqualTo(8090);
178         * </code>
179         * </pre>
180         * 
181         * This is a shorthand method that should be used if only one {@link Argument} is expected as it
182         * will throw if unexpected arguments are encountered. If several arguments are expected use
183         * {@link CommandLineParser#withArguments(Argument...)} instead.
184         * 
185         * @param actualArguments the arguments from the command line
186         * @return the parsed value from the {@code actualArguments}
187         * @throws ArgumentException if actualArguments isn't compatible with this
188         *             argument
189         *             </pre>
190         */
191        @Nullable
192        public final T parse(String ... actualArguments) throws ArgumentException
193        {
194                return build().parse(actualArguments);
195        }

198         * Returns a usage string for the argument that this builder builds.

200         * otherwise the {@link CommandLineParser#usage()} method should be used instead.
201         */
202        public final Usage usage()
203        {
204                return build().usage();
205        }

207        /**
208         * <pre>
209         * Returns a customized parser that the {@link Argument} will use to parse values.
210         * 
211         * This is a suitable place to verify the configuration of your parser.
212         * 
213         * If your {@link StringParser} doesn't support any configuration you can use
214         * {@link Arguments#withParser(StringParser)} directly instead of subclassing
215         * {@link ArgumentBuilder}
216         * 
217         * @return the {@link StringParser} that performs the actual parsing of an argument value
218         * @throws IllegalStateException if the parser have been configured wrongly
219         * </pre>
220         */
221        @Nonnull
222        protected abstract StringParser<T> parser();
223
224        @Nonnull
225        InternalStringParser<T> internalParser()
226        {
227                StringParser<T> parser = parser();
228                if(parser != InternalArgumentBuilder.MARKER)
229                        return new StringParserBridge<T>(parser);
230                return internalStringParser;
231        }

327         * <pre>
328         * Makes {@link CommandLineParser#parse(String...)} throw
329         * {@link ArgumentException} if this argument isn't given.
330         * If possible, it's preferred to use a {@link #defaultValue(Object) default value} instead.
331         * 
332         * The {@link Argument#toString()} will be used to print each missing argument.
333         * 
334         * <b>Note</b>: If you choose to use multiple {@link #required()} indexed arguments all of them
335         * must have unique {@link #metaDescription(String)}s. This ensures that error messages better
336         * can point out erroneous arguments
337         * 
338         * @return this builder
339         * @throws IllegalStateException if {@link #defaultValue(Object)} (or
340         *             {@link #defaultValueSupplier(Supplier)}) has been
341         *             called, because these are mutually exclusive with {@link #required()}
342         * </pre>

347                required = true;
348                return myself;

351        /**
352         * <pre>
353         * Sets a default value to use for this argument. Overrides {@link StringParser#defaultValue()} which is used by default.
354         * Returned by {@link ParsedArguments#get(Argument)} when no argument {@link ParsedArguments#wasGiven(Argument) was given}.
355         * To create default values lazily see {@link ArgumentBuilder#defaultValueSupplier(Supplier)}.
356         * 
357         * <b>Mutability</b>:Remember that as {@link Argument} is {@link Immutable}
358         * this value should be so too if multiple argument parsings is going to take place.
359         * If mutability is wanted {@link ArgumentBuilder#defaultValueSupplier(Supplier)} should be used instead.
360         * 
361         * @return this builder
362         * @throws IllegalStateException if {@link #required()} has been called,
363         * because these two methods are mutually exclusive
364         */
365        public SELF defaultValue(@Nullable final T value)
366        {
367                checkState(!required, ProgrammaticErrors.DEFAULT_VALUE_AND_REQUIRED);
368                defaultValueSupplier = Suppliers.ofInstance(value);
369                return myself;
370        }

373         * <pre>
374         * Sets a {@link Supplier} that can supply default values in the absence of this argument
375         * 
376         * <b>Note:</b> Even if {@link #limitTo(Predicate)} is used, the {@link Supplier#get()} isn't called
377         * until the default value is actually needed ({@link ParsedArguments#get(Argument)}. If the
378         * default value is deemed non-allowed at that point an {@link IllegalStateException} is thrown.
379         * 
380         * <b>Note:</b> May be removed in the future if Guava is removed as a dependency
381         * 
382         * Wrap your supplier with {@link Suppliers#memoize(Supplier)} if you want to cache created values.
383         * 
384         * @return this builder
385         * @throws IllegalStateException if {@link #required()} has been called,
386         * because these two methods are mutually exclusive
387         */
388        @Beta
389        public SELF defaultValueSupplier(final Supplier<? extends T> aDefaultValueSupplier)
390        {
391                checkState(!required, ProgrammaticErrors.DEFAULT_VALUE_AND_REQUIRED);
392                defaultValueSupplier = checkNotNull(aDefaultValueSupplier);
393                return myself;
394        }

423        /**
424         * <pre>
425         * By default {@link StringParser}s provides a meta description (by implementing {@link StringParser#metaDescription()}
426         * that describes the type of data they expect. For instance, if you're writing a music player,
427         * a user of your application would (most probably) rather see:
428         * 
429         * --track-nr &lt;track nr&gt;    The track number to play (using {@code  metaDescription("<track nr>") })
430         * 
431         * instead of
432         * 
433         * --track-nr &lt;integer&gt;     The track number to play (the default provided by {@link StringParser#metaDescription()})
434         * 
435         * So when using general data type parsers such as {@link StringParsers#integerParser()} you're better of if
436         * you provide a meta description that explains what the {@code integer} represents in the context of this argument.
437         * 
438         * <b>Note:</b> empty meta descriptions aren't allowed
439         * <b>Note:</b> the surrounding < & > aren't enforced or added automatically but it's preferred
440         * to have them because it makes a clear distinction between {@link #names(String...) argument names} and their parameters.
441         * 
442         * @param aMetaDescription "&lt;track nr&gt;" in the above example
443         * @return this builder
444         * @throws IllegalArgumentException if aMetaDescription is empty
445         * </pre>
446         */
447        public final SELF metaDescription(final String aMetaDescription)
448        {
449                checkArgument(aMetaDescription.length() > 0, ProgrammaticErrors.INVALID_META_DESCRIPTION);
450                this.metaDescription = Optional.of(aMetaDescription);
451                return myself;
452        }

481         * <pre>
482         * Limits values parsed so that they conform to some specific rule.
483         * Only values for which {@link Predicate#apply(Object)} returns true, will be accepted.
484         * Other values will cause an {@link ArgumentException}.
485         * For example {@link Range#closed(Comparable, Comparable)} only allows values within a range.
486         * 
487         * To override the default error message that is generated with {@link UserErrors#DISALLOWED_VALUE}
488         * you can throw {@link IllegalArgumentException} from {@link Predicate#apply(Object)}. The detail
489         * message of that exception will be used by {@link ArgumentException#getMessageAndUsage()}.
490         * When this is needed it's generally recommended to write a parser of its own instead.
491         * 
492         * <b>Note:</b>{@link Object#toString() toString()} on {@code aLimiter} will replace {@link StringParser#descriptionOfValidValues(Locale)} in the usage
493         * 
494         * <b>Note:</b>The validity of any {@link #defaultValueSupplier(Supplier) default value} isn't checked until
495         * it's actually needed when {@link ParsedArguments#get(Argument)} is called. This is so
496         * because {@link Supplier#get()} (or {@link StringParser#defaultValue()}) could take an arbitrary long time.
497         * </pre>
498         * 
499         * @param aLimiter a limiter
500         * @return this builder
501         */

505                limiter = checkNotNull(aLimiter);

509        /**
510         * <pre>
511         * Makes this argument handle "property like" arguments:
512         * -Dproperty_name=value
513         * where
514         * "-D" is the string supplied to {@link #names(String...)},
515         * "property_name" is the {@link String} key in the resulting {@link Map},
516         * "=" is the {@link #separator(String)} (set to "=" if it hasn't been overridden already),
517         * "value" is decoded by the previously set {@link StringParser}.
518         * 
519         * 
520         * Tip: You can pass {@link #defaultValueDescriber(Describer)} a
521         * {@link Describers#mapDescriber(Map)} to describe each property that you
522         * support (given that you've used {@link ArgumentBuilder#defaultValue(Object)}
523         * to use sane defaults for your properties).
524         * 
525         * @return a new (more specific) builder
526         * @throws IllegalStateException if {@link #defaultValue(Object)} has been set before {@link #asPropertyMap()} is called
527         * 
528         * @see #asKeyValuesWithKeyParser(StringParser) to use other types as keys than {@link String}
529         * </pre>
530         */
531        @CheckReturnValue
532        public final MapArgumentBuilder<String, T> asPropertyMap()
533        {
534                checkState(defaultValueSupplier == null, ProgrammaticErrors.INVALID_CALL_ORDER, "defaultValue", "asPropertyMap");
535                return new MapArgumentBuilder<String, T>(this, stringParser());
536        }

538        /**
539         * <pre>
540         * Makes this argument handle properties like arguments:
541         * -Dproperty_name=value
542         * where "-D" is one of the strings supplied to {@link #names(String...)},
543         * "property_name" is decoded by {@code keyParser} and
544         * "value" is decoded by the {@link StringParser} previously passed to the constructor.
545         * 
546         * For example:
547         * </pre>
548         * 
549         * <pre class="prettyprint">
550         * <code class="language-java">
551         * Map&lt;Integer, Integer&gt; numberMap = Arguments.integerArgument("-N")
552         *                                              .asKeyValuesWithKeyParser(StringParsers.integerParser())
553         *                                              .parse("-N1=5", "-N2=10");
554         * assertThat(numberMap.get(1)).isEqualTo(5);
555         * </code>
556         * </pre>
557         * 
558         * For this to work correctly it's paramount that {@code K} implements a
559         * proper {@link Object#hashCode()} because it's going to be used a key in a {@link Map}.
560         * 
561         * @return a new (more specific) builder
562         * @throws IllegalStateException if a {@link #defaultValueDescriber(Describer)} or
563         *             {@link #defaultValue(Object)} has been set as they have no place in a property
564         *             map.
565         */
566        @CheckReturnValue
567        public final <K> MapArgumentBuilder<K, T> asKeyValuesWithKeyParser(StringParser<K> keyParser)
568        {
569                checkState(defaultValueSupplier == null, ProgrammaticErrors.INVALID_CALL_ORDER, "defaultValue", "asKeyValuesWithKeyParser");
570                return new MapArgumentBuilder<K, T>(this, checkNotNull(keyParser));
571        }

629         * <pre class="prettyprint">
630         * <code class="language-java">
631         * List&lt;Integer&gt; numbers = integerArgument("--numbers").arity(2).parse("--numbers", "1", "2");

633         * </code>
634         * </pre>
635         * 
636         * <b>Note:</b>If the argument isn't {@link #required()} the default value
637         * will be a list that contains {@code numberOfParameters} elements
638         * of {@link StringParser#defaultValue()}, in the above example that would be two zeros.
639         * If this isn't wanted use {@link #defaultValue(Object)} to override it.
640         * 
641         * @return a new (more specific) builder
642         * @throws IllegalArgumentException if {@code numberOfParameters} is less than 2
643         */
644        @CheckReturnValue
645        public ArityArgumentBuilder<T> arity(final int numberOfParameters)
646        {
647                checkArgument(numberOfParameters > 1, ProgrammaticErrors.TO_SMALL_ARITY, numberOfParameters);
648                return new ArityArgumentBuilder<T>(this, numberOfParameters);
649        }
