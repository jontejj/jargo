044/**
045 * <pre>
046 * Represents a supported {@link Argument} for a command line invocation.
047 * 
048 * {@link Argument}s are created with the static methods in {@link Arguments} or with a custom {@link ArgumentBuilder} and then
049 * used by the {@link CommandLineParser} to parse strings (typically from the command line).
050 * 
051 * @param <T> the type of values this {@link Argument} is configured to parse
052 * </pre>
053 */
054@Immutable
055public final class Argument<T>
056{
057        @Nonnull private final List<String> names;
058        @Nonnull private final Describable description;
059        @Nonnull private final Optional<String> metaDescription;
060        private final boolean hideFromUsage;
061
062        @Nonnull private final String separator;
063        private final boolean required;
064        private final boolean ignoreCase;
065        private final boolean isAllowedToRepeat;
066
067        @Nonnull private final InternalStringParser<T> parser;
068        @Nonnull private final Supplier<? extends T> defaultValue;
069        @Nullable private final Describer<? super T> defaultValueDescriber;
070        @Nonnull private final Predicate<? super T> limiter;
071
072        // Internal bookkeeping
073        @Nonnull private final Function<T, T> finalizer;
074        private final boolean isPropertyMap;
075
076        private final CollationKey sortingKey;
077        private static final Collator LINGUISTIC_ORDER = Collator.getInstance(Locale.ROOT);
078
079        /**
080         * <pre>
081         * Creates a basic object for handling {@link Argument}s taken from a command line invocation.
082         * For practical uses of this constructor see {@link Arguments#optionArgument(String, String...)} (and friends)
083         * and the {@link ArgumentBuilder}.
084         * </pre>
085         */
086        Argument(final ArgumentBuilder<?, T> builder)
087        {
088                this.parser = builder.internalParser();
089                this.defaultValueDescriber = builder.defaultValueDescriber();
090                this.description = builder.description();
091                this.required = builder.isRequired();
092                this.separator = builder.separator();
093                this.ignoreCase = builder.isIgnoringCase();
094                this.names = builder.names();
095                this.isPropertyMap = builder.isPropertyMap();
096                this.isAllowedToRepeat = builder.isAllowedToRepeat();
097                this.hideFromUsage = builder.isHiddenFromUsage();
098                this.metaDescription = builder.metaDescription();
099
100                this.finalizer = builder.finalizer();
101                this.limiter = builder.limiter();
102                this.defaultValue = builder.defaultValueSupplierOrFromParser();
103
104                // Fail-fast for invalid default values that are created already
105                if(Suppliers2.isSuppliedAlready(defaultValue))
106                {
107                        // Calling this makes sure that the default value is within the limits of any limiter
108                        defaultValue();
109                }
110
111                // Better to take the hit up front than to lazy load and experience latency when usage is
112                // requested for the first time
113                this.sortingKey = LINGUISTIC_ORDER.getCollationKey(toString());
114        }
115
116        /**
117         * Parses command line arguments and returns the value of this argument.<br>
118         * This is a shorthand method that should be used if only one {@link Argument} is expected.
119         * If several arguments are expected use {@link CommandLineParser#withArguments(Argument...)}
120         * instead. <br>
121         * <b>Locale:</b> {@link Locale#US} is used to parse strings / print usage. To use another
122         * {@link Locale} you'll need to use {@link CommandLineParser#withArguments(Argument...)}
123         * instead.
124         * 
125         * @param actualArguments the arguments from the command line
126         * @return the parsed value from the {@code actualArguments}
127         * @throws ArgumentException if actualArguments isn't what this
128         *             argument expects
129         */
130        @Nullable
131        public T parse(String ... actualArguments) throws ArgumentException
132        {
133                return commandLineParser().parse(asList(actualArguments)).get(this);
134        }
135
136        /**
137         * Returns the {@link Usage} for this argument. Should only be used if one argument is
138         * supported, otherwise the {@link CommandLineParser#usage()} method should be used instead.
139         */
140        @Nonnull
141        @CheckReturnValue
142        public Usage usage()
143        {
144                return new Usage(commandLineParser());
145        }
146
147        /**
148         * Describes {@link Argument}s by their first name. If they are indexed and have no name the
149         * {@link StringParser#metaDescription() meta description} is used instead.
150         */
151        @Override
152        public String toString()
153        {
154                if(isIndexed())
155                        return metaDescriptionInRightColumn();
156                return names().get(0);
157        }
158
159        /**
160         * @return the default value for this argument, defaults to {@link StringParser#defaultValue()}.
161         *         Could also be set by {@link ArgumentBuilder#defaultValue(Object)} or
162         *         {@link ArgumentBuilder#defaultValueSupplier(Supplier)}
163         */
164        @Nullable
165        T defaultValue()
166        {
167                T value = finalizer.apply(defaultValue.get());
168
169                // If this throws it indicates that a bad combination of Provider/Limiter
170                // is used or simply put that it's an invalid default value
171                checkLimitForDefaultValue(value);
172
173                return value;
174        }
175
176        String descriptionOfValidValues(Locale inLocale)
177        {
178                if(limiter != alwaysTrue())
179                        return limiter.toString();
180
181                return parser().descriptionOfValidValues(this, inLocale);
182        }
183
184        @Nullable
185        String defaultValueDescription(Locale inLocale)
186        {
187                T value = defaultValue();
188                if(defaultValueDescriber != null)
189                        return defaultValueDescriber.describe(value, inLocale);
190                return parser().describeValue(value);
191        }
192
193        @Nonnull
194        String metaDescriptionInLeftColumn()
195        {
196                String meta = metaDescription.orElseGet(() -> parser.metaDescriptionInLeftColumn(this));
197                if(!isPropertyMap() && !isIndexed())
198                {
199                        meta = separator + meta;
200                }
201                return meta;
202        }
203
204        @Nonnull
205        String metaDescriptionInRightColumn()
206        {
207                return metaDescription.orElseGet(() -> parser.metaDescriptionInRightColumn(this));
208        }
209
210        void checkLimit(@Nullable final T value) throws ArgumentException
211        {
212                if(!limiter.test(value))
213                        throw withMessage(format(UserErrors.DISALLOWED_VALUE, value, limiter));
214        }
215
216        private void checkLimitForDefaultValue(@Nullable final T value)
217        {
218                boolean error = false;
219                try
220                {
221                        if(!limiter.test(value))
222                        {
223                                error = true;
224                        }
225                }
226                catch(IllegalArgumentException invalidDefaultValue)
227                {
228                        throw new IllegalArgumentException(String.format(ProgrammaticErrors.INVALID_DEFAULT_VALUE, invalidDefaultValue.getMessage()),
229                                        invalidDefaultValue);
230                }
231                if(error)
232                {
233                        String disallowedValue = String.format(UserErrors.DISALLOWED_VALUE, value, limiter);
234                        throw new IllegalArgumentException(String.format(ProgrammaticErrors.INVALID_DEFAULT_VALUE, disallowedValue));
235                }
236        }
237
238        @Nonnull
239        InternalStringParser<T> parser()
240        {
241                return parser;
242        }
243
244        boolean isRequired()
245        {
246                return required;
247        }
248
249        @Nullable
250        String separator()
251        {
252                return separator;
253        }
254
255        @Nonnull
256        String description()
257        {
258                return description.description();
259        }
260
261        @Nonnull
262        List<String> names()
263        {
264                return names;
265        }
266
267        boolean isPropertyMap()
268        {
269                return isPropertyMap;
270        }
271
272        boolean isAllowedToRepeat()
273        {
274                return isAllowedToRepeat;
275        }
276
277        boolean isIgnoringCase()
278        {
279                return ignoreCase;
280        }
281
282        Function<T, T> finalizer()
283        {
284                return finalizer;
285        }
286
287        boolean isIndexed()
288        {
289                return names().isEmpty();
290        }
291
292        boolean isHelpArgument()
293        {
294                return parser().equals(HelpParser.INSTANCE);
295        }
296
297        private CommandLineParserInstance commandLineParser()
298        {
299                // Not cached to save memory, users should use CommandLineParser.withArguments if they are
300                // concerned about reuse
301                return new CommandLineParserInstance(Arrays.<Argument<?>>asList(Argument.this));
302        }
303
304        /**
305         * Classes of how many arguments that can be handled by an {@link Argument}.
306         */
307        enum ParameterArity
308        {
309                /**
310                 * Indicates {@link Arguments#optionArgument(String, String...)}
311                 */
312                NO_ARGUMENTS,
313                /**
314                 * Indicates {@link ArgumentBuilder#variableArity()}
315                 */
316                VARIABLE_AMOUNT,
317                /**
318                 * {@link ArgumentBuilder#arity(int)} or any other {@link Argument}
319                 */
320                AT_LEAST_ONE_ARGUMENT
321        }
322
323        static final Predicate<Argument<?>> IS_INDEXED = Argument::isIndexed;
324
325        static final Predicate<Argument<?>> IS_REQUIRED = Argument::isRequired;
326
327        static final Predicate<Argument<?>> IS_VISIBLE = input -> !input.hideFromUsage;
328
329        static final Predicate<Argument<?>> IS_REPEATED = input -> input.isAllowedToRepeat;
330
331        static final Predicate<Argument<?>> IS_OF_VARIABLE_ARITY = input -> input.parser().parameterArity() == ParameterArity.VARIABLE_AMOUNT;
332
333        // TODO(jontejj): replace this with a comparator that uses the Usage.locale instead of
334        // Locale.ROOT?
335        static final Comparator<Argument<?>> NAME_COMPARATOR = (lhs, rhs) -> lhs.sortingKey.compareTo(rhs.sortingKey);
336}
