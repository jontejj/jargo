048/**
049 * <pre>
050 * Represents a supported {@link Argument} for a command line invocation.
051 * 
052 * {@link Argument}s are created with the static methods in {@link Arguments} or with a custom {@link ArgumentBuilder} and then
053 * used by the {@link CommandLineParser} to parse strings (typically from the command line).
054 * 
055 * @param <T> the type of values this {@link Argument} is configured to parse
056 * 
057 * </pre>
058 */
059@Immutable
060public final class Argument<T>
061{
062        @Nonnull private final List<String> names;
063        @Nonnull private final Describable description;
064        @Nonnull private final Optional<String> metaDescription;
065        private final boolean hideFromUsage;
066
067        @Nonnull private final String separator;
068        @Nonnull private final Optional<Locale> localeOveride;
069        private final boolean required;
070        private final boolean ignoreCase;
071        private final boolean isAllowedToRepeat;
072
073        @Nonnull private final InternalStringParser<T> parser;
074        @Nonnull private final Supplier<? extends T> defaultValue;
075        @Nullable private final Describer<? super T> defaultValueDescriber;
076        @Nonnull private final Predicate<? super T> limiter;
077
078        // Internal bookkeeping
079        @Nonnull private final Function<T, T> finalizer;
080        private final ParameterArity parameterArity;
081        private final boolean isPropertyMap;
082
083        private final CollationKey sortingKey;
084        private static final Collator LINGUISTIC_ORDER = Collator.getInstance(Locale.ROOT);
085
086        /**
087         * <pre>
088         * Creates a basic object for handling {@link Argument}s taken from a command line invocation.
089         * For practical uses of this constructor see {@link Arguments#optionArgument(String, String...)} (and friends)
090         * and the {@link ArgumentBuilder}.
091         * </pre>
092         */
093        Argument(final ArgumentBuilder<?, T> builder)
094        {
095                this.parser = builder.internalParser();
096                this.defaultValueDescriber = builder.defaultValueDescriber();
097                this.description = builder.description();
098                this.required = builder.isRequired();
099                this.separator = builder.separator();
100                this.localeOveride = builder.localeOverride();
101                this.ignoreCase = builder.isIgnoringCase();
102                this.names = builder.names();
103                this.isPropertyMap = builder.isPropertyMap();
104                this.isAllowedToRepeat = builder.isAllowedToRepeat();
105                this.hideFromUsage = builder.isHiddenFromUsage();
106                this.metaDescription = builder.metaDescription();
107                this.parameterArity = builder.parameterArity();
108
109                this.finalizer = builder.finalizer();
110                this.limiter = builder.limiter();
111                if(builder.defaultValueSupplier() != null)
112                {
113                        this.defaultValue = builder.defaultValueSupplier();
114                }
115                else
116                {
117                        this.defaultValue = new Supplier<T>(){
118                                @Override
119                                public T get()
120                                {
121                                        return parser.defaultValue();
122                                }
123                        };
124                }
125
126                // Fail-fast for invalid default values that are created already
127                if(Suppliers2.isSuppliedAlready(defaultValue))
128                {
129                        // Calling this makes sure that the default value is within the limits of any limiter
130                        defaultValue();
131                }
132
133                // Better to take the hit up front than to lazy load and experience latency when usage is
134                // requested for the first time
135                this.sortingKey = LINGUISTIC_ORDER.getCollationKey(toString());
136        }
137
138        /**
139         * Parses command line arguments and returns the value of this argument.<br>
140         * This is a shorthand method that should be used if only one {@link Argument} is expected.
141         * If several arguments are expected use {@link CommandLineParser#withArguments(Argument...)}
142         * instead. {@link Locale#US} is used by default to parse strings. Use
143         * {@link ArgumentBuilder#locale(Locale)} to specify another {@link Locale}, such as
144         * {@link Locale#getDefault()}.
145         * 
146         * @param actualArguments the arguments from the command line
147         * @return the parsed value from the {@code actualArguments}
148         * @throws ArgumentException if actualArguments isn't what this
149         *             argument expects
150         */
151        @Nullable
152        public T parse(String ... actualArguments) throws ArgumentException
153        {
154                return commandLineParser().parse(asList(actualArguments), locale(US_BY_DEFAULT)).get(this);
155        }
156
157        /**
158         * Returns the {@link Usage} for this argument. Should only be used if one argument is
159         * supported, otherwise the {@link CommandLineParser#usage()} method should be used instead.
160         * {@link Locale#US} is used by default. Use {@link ArgumentBuilder#locale(Locale)} to specify
161         * another {@link Locale}, such as {@link Locale#getDefault()}.
162         */
163        @Nonnull
164        @CheckReturnValue
165        public Usage usage()
166        {
167                return new Usage(commandLineParser().allArguments(), locale(US_BY_DEFAULT), ProgramInformation.AUTO, false);
168        }
169
170        /**
171         * Describes {@link Argument}s by their first name. If they are indexed and have no name the
172         * {@link StringParser#metaDescription() meta description} is used instead.
173         */
174        @Override
175        public String toString()
176        {
177                if(isIndexed())
178                        return metaDescriptionInRightColumn();
179                return names().get(0);
180        }
181
182        /**
183         * @return the default value for this argument, defaults to {@link StringParser#defaultValue()}.
184         *         Could also be set by {@link ArgumentBuilder#defaultValue(Object)} or
185         *         {@link ArgumentBuilder#defaultValueSupplier(Supplier)}
186         */
187        @Nullable
188        T defaultValue()
189        {
190                T value = finalizer.apply(defaultValue.get());
191
192                // If this throws it indicates that a bad combination of Provider/Limiter
193                // is used or simply put that it's an invalid default value
194                checkLimitForDefaultValue(value);
195
196                return value;
197        }
198
199        String descriptionOfValidValues(Locale localeToDescribeValuesWith)
200        {
201                if(limiter != alwaysTrue())
202                        return limiter.toString();
203
204                return parser().descriptionOfValidValues(this, locale(localeToDescribeValuesWith));
205        }
206
207        @Nullable
208        String defaultValueDescription(Locale inLocale)
209        {
210                T value = defaultValue();
211                // TODO(jontejj): use locale(inLocale) here
212                if(defaultValueDescriber != null)
213                        return defaultValueDescriber.describe(value, inLocale);
214                return parser().describeValue(value);
215        }
216
217        @Nonnull
218        String metaDescriptionInLeftColumn()
219        {
220                String meta = metaDescription.or(parser.metaDescriptionInLeftColumn(this));
221                if(!isPropertyMap() && !isIndexed())
222                {
223                        meta = separator + meta;
224                }
225                return meta;
226        }
227
228        @Nonnull
229        String metaDescriptionInRightColumn()
230        {
231                return metaDescription.or(parser.metaDescriptionInRightColumn(this));
232        }
233
234        void checkLimit(@Nullable final T value) throws ArgumentException
235        {
236                if(!limiter.apply(value))
237                        throw withMessage(format(UserErrors.DISALLOWED_VALUE, value, limiter));
238        }
239
240        private void checkLimitForDefaultValue(@Nullable final T value)
241        {
242                try
243                {
244                        if(!limiter.apply(value))
245                        {
246                                String unallowedValue = String.format(UserErrors.DISALLOWED_VALUE, value, limiter);
247                                throw new IllegalStateException(String.format(ProgrammaticErrors.INVALID_DEFAULT_VALUE, unallowedValue));
248                        }
249                }
250                catch(IllegalArgumentException invalidDefaultValue)
251                {
252                        throw new IllegalStateException(String.format(ProgrammaticErrors.INVALID_DEFAULT_VALUE, invalidDefaultValue.getMessage()),
253                                        invalidDefaultValue);
254                }
255        }
256
257        @Nonnull
258        InternalStringParser<T> parser()
259        {
260                return parser;
261        }
262
263        boolean isRequired()
264        {
265                return required;
266        }
267
268        @Nullable
269        String separator()
270        {
271                return separator;
272        }
273
274        @Nonnull
275        String description()
276        {
277                return description.description();
278        }
279
280        @Nonnull
281        List<String> names()
282        {
283                return names;
284        }
285
286        boolean isPropertyMap()
287        {
288                return isPropertyMap;
289        }
290
291        boolean isAllowedToRepeat()
292        {
293                return isAllowedToRepeat;
294        }
295
296        ParameterArity parameterArity()
297        {
298                return parameterArity;
299        }
300
301        boolean isIgnoringCase()
302        {
303                return ignoreCase;
304        }
305
306        Locale locale(Locale usualLocale)
307        {
308                return localeOveride.or(usualLocale);
309        }
310
311        Function<T, T> finalizer()
312        {
313                return finalizer;
314        }
315
316        boolean isIndexed()
317        {
318                return names().isEmpty();
319        }
320
321        boolean isHelpArgument()
322        {
323                return parser().equals(HelpParser.INSTANCE);
324        }
325
326        private CommandLineParserInstance commandLineParser()
327        {
328                // Not cached to save memory, users should use CommandLineParser.withArguments if they are
329                // concerned about reuse
330                return new CommandLineParserInstance(Arrays.<Argument<?>>asList(Argument.this), ProgramInformation.AUTO);
331        }
332
333        enum ParameterArity
334        {
335                /**
336                 * Indicates {@link Arguments#optionArgument(String, String...)}
337                 */
338                NO_ARGUMENTS,
339                /**
340                 * Indicates {@link ArgumentBuilder#variableArity()}
341                 */
342                VARIABLE_AMOUNT,
343                /**
344                 * {@link ArgumentBuilder#arity(int)} or any other {@link Argument}
345                 */
346                AT_LEAST_ONE_ARGUMENT
347        }
348
349        static final Predicate<Argument<?>> IS_INDEXED = new Predicate<Argument<?>>(){
350                @Override
351                public boolean apply(@Nonnull Argument<?> input)
352                {
353                        return input.isIndexed();
354                }
355        };
356
357        static final Predicate<Argument<?>> IS_REQUIRED = new Predicate<Argument<?>>(){
358                @Override
359                public boolean apply(@Nonnull Argument<?> input)
360                {
361                        return input.isRequired();
362                }
363        };
364
365        static final Predicate<Argument<?>> IS_VISIBLE = new Predicate<Argument<?>>(){
366                @Override
367                public boolean apply(@Nonnull Argument<?> input)
368                {
369                        return !input.hideFromUsage;
370                }
371        };
372
373        static final Predicate<Argument<?>> IS_OF_VARIABLE_ARITY = new Predicate<Argument<?>>(){
374                @Override
375                public boolean apply(@Nonnull Argument<?> input)
376                {
377                        return input.parameterArity() == ParameterArity.VARIABLE_AMOUNT;
378                }
379        };
380
381        static final Comparator<Argument<?>> NAME_COMPARATOR = new Comparator<Argument<?>>(){
382                // TODO(jontejj): replace this with a comparator that uses the Usage.locale instead of
383                // Locale.ROOT?
384                @Override
385                public int compare(Argument<?> lhs, Argument<?> rhs)
386                {
387                        return lhs.sortingKey.compareTo(rhs.sortingKey);
388                }
389        };
