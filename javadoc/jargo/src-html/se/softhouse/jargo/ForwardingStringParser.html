015package se.softhouse.jargo;
016
017import java.util.Locale;
018
019import javax.annotation.Nonnull;
020import javax.annotation.concurrent.Immutable;
021
022/**
023 * A <a href="http://en.wikipedia.org/wiki/Decorator_pattern">Decorator</a> class.
024 * It allows you to subclass it and override individual methods
025 * that you want to customize for an existing {@link StringParser}.
026 *
027 * For instance, a parser that's useful in a garden application could look like this:
028 *
029 * <pre class="prettyprint">
030 * <code class="language-java">
031 * private static final class WateringParser extends SimpleForwardingStringParser&lt;Integer&gt;
032 * {
033 *   WateringParser()
034 *   {
035 *     super(StringParsers.integerParser());
036 *   }
037 *
038 *   public Integer parse(String value, Locale locale) throws ArgumentException
039 *   {
040 *     waterPlants();
041 *     return super.parse(value, locale);
042 *   }
043 *
044 *   private void waterPlants()
045 *   {
046 *     System.out.println("Watering plants");
047 *   }
048 * }
049 * </code>
050 *
051 * This WateringParser can then be integrated with an argument via
052 * {@link Arguments#withParser(StringParser)}.
053 * Most subclasses can just use {@link SimpleForwardingStringParser}.
054 * </pre>
055 *
056 * @param <T> the type the decorated {@link StringParser} handles
057 */
058@Immutable
059public abstract class ForwardingStringParser<T> implements StringParser<T>
060{
061        /**
062         * A factory method that allow subclasses to switch delegate under their own terms,
063         * like the <a href="http://en.wikipedia.org/wiki/Proxy_pattern">Proxy</a> pattern explains.
064         * Just remember that a {@link StringParser} should be treated like it's {@link Immutable} so
065         * make sure your callers can't tell the difference when you switch parser.
066         *
067         * @return the delegate to pass non-overridden calls to
068         */
069        @Nonnull
070        protected abstract StringParser<T> delegate();
071
072        @Override
073        public String descriptionOfValidValues(Locale locale)
074        {
075                return delegate().descriptionOfValidValues(locale);
076        }
077
078        @Override
079        public T parse(final String value, Locale locale) throws ArgumentException
080        {
081                return delegate().parse(value, locale);
082        }
083
084        @Override
085        public T defaultValue()
086        {
087                return delegate().defaultValue();
088        }
089
090        @Override
091        public String metaDescription()
092        {
093                return delegate().metaDescription();
094        }
095
096        @Override
097        public String toString()
098        {
099                return descriptionOfValidValues(Locale.US);
100        }
101
102        /**
103         * A {@link ForwardingStringParser} that uses an already created {@link StringParser} as its
104         * delegate.
105         *
106         * @param <T> the type the decorated {@link StringParser} handles
107         */
108        public abstract static class SimpleForwardingStringParser<T> extends ForwardingStringParser<T>
109        {
110                @Nonnull private final StringParser<T> delegate;
111
112                protected SimpleForwardingStringParser(final StringParser<T> delegate)
113                {
114                        this.delegate = delegate;
115                }
116
117                @Override
118                protected final StringParser<T> delegate()
119                {
120                        return delegate;
121                }
122        }
123}
