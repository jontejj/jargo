252        /**
253         * <pre>
254         * A parser that uses {@link Enum#valueOf(Class, String)} to {@link StringParser#parse(String, Locale) parse} input strings.
255         *
256         * <b>Case sensitivity note:</b> First a direct match with the input value in upper case is
257         * made, if that fails a direct match without converting the case is made,
258         * if that also fails an {@link ArgumentException} is thrown. This order of execution is
259         * based on the fact that users typically don't upper case their input while
260         * <a href=
261        "http://www.oracle.com/technetwork/java/javase/documentation/codeconventions-135099.html#367">java naming conventions</a> recommends upper case for enum constants.
262         * </pre>
263         *
264         * <b>Default value:</b> <code>null</code> is used as {@link StringParser#defaultValue()} and
265         * not any of the enum values in {@code enumToHandle}
266         *
267         * @param enumToHandle the {@link Class} literal for the {@link Enum} to parse strings into
268         * @return a {@link StringParser} that parses strings into enum values of the type {@code T}
269         */
270        @CheckReturnValue
271        @Nonnull
272        public static <T extends Enum<T>> StringParser<T> enumParser(final Class<T> enumToHandle)
273        {
274                return new EnumParser<T>(enumToHandle);
275        }
276
277        private static final class EnumParser<E extends Enum<E>> implements StringParser<E>
278        {
279                private final Class<E> enumType;
280
281                private EnumParser(final Class<E> enumToHandle)
282                {
283                        enumType = requireNonNull(enumToHandle);
284                }
285
286                @Override
287                public E parse(final String value, final Locale locale) throws ArgumentException
288                {
289                        try
290                        {
291                                return Enum.valueOf(enumType, value.toUpperCase(Locale.US));
292                        }
293                        catch(IllegalArgumentException noEnumFoundWithUpperCase)
294                        {
295                                try
296                                {
297                                        // Try an exact match just in case an enum value doesn't follow java naming
298                                        // guidelines
299                                        return Enum.valueOf(enumType, value);
300                                }
301                                catch(IllegalArgumentException noEnumFoundWithExactMatch)
302                                {
303                                        throw withMessage(format(UserErrors.INVALID_ENUM_VALUE, value, new Object(){
304                                                @Override
305                                                public String toString()
306                                                {
307                                                        // Lazily call this as it's going over all enum values and converting
308                                                        // them to strings
309                                                        return descriptionOfValidValues(locale);
310                                                }
311                                        }), noEnumFoundWithExactMatch);
312                                }
313                        }
314                }
315
316                @Override
317                public String descriptionOfValidValues(Locale locale)
318                {
319                        E[] enumValues = enumType.getEnumConstants();
320                        StringJoiner joiner = new StringJoiner(" | ", "{", "}");
321                        Arrays.stream(enumValues).forEach(e -> joiner.add(e.toString()));
322                        return joiner.toString();
323                }
324
325                @Override
326                public E defaultValue()
327                {
328                        return null;
329                }
330
331                @Override
332                public String metaDescription()
333                {
334                        return "<" + enumType.getSimpleName() + ">";
335                }
336        }

454        static final class TransformingParser<T, F> extends InternalStringParser<F>
455        {
456                private final Function<T, F> transformer;
457                private final InternalStringParser<T> firstParser;
458                private final Predicate<? super T> limiter;
459
460                TransformingParser(InternalStringParser<T> firstParser, Function<T, F> transformer, Predicate<? super T> limiter)
461                {
462                        this.firstParser = requireNonNull(firstParser);
463                        this.transformer = requireNonNull(transformer);
464                        this.limiter = requireNonNull(limiter);
465                }
466
467                @SuppressWarnings("unchecked")
468                @Override
469                F parse(ArgumentIterator arguments, F previousOccurance, Argument<?> argumentSettings, Locale locale) throws ArgumentException
470                {
471                        T first = firstParser.parse(arguments, null, argumentSettings, locale);
472                        if(!limiter.test(first))
473                                throw withMessage(format(UserErrors.DISALLOWED_VALUE, first, argumentSettings.descriptionOfValidValues(locale)));
474
475                        F result = transformer.apply(first);
476                        if(previousOccurance instanceof Collection && result instanceof Collection && argumentSettings.isAllowedToRepeat())
477                        {
478                                ((Collection<Object>) previousOccurance).addAll((Collection<Object>) result);
479                                return previousOccurance;
480                        }
481                        return result;
482                }
483
484                @Override
485                String descriptionOfValidValues(Argument<?> argumentSettings, Locale locale)
486                {
487                        if(limiter != Predicates2.alwaysTrue())
488                        {
489                                String attemptAtGoodDescription = limiter.toString();
490                                if(!attemptAtGoodDescription.contains("$$Lambda$"))
491                                        return attemptAtGoodDescription;
492                        }
493                        return firstParser.descriptionOfValidValues(argumentSettings, locale);
494                }
495
496                @Override
497                String metaDescriptionInLeftColumn(Argument<?> argumentSettings)
498                {
499                        return firstParser.metaDescriptionInLeftColumn(argumentSettings);
500                }
501
502                @Override
503                String metaDescriptionInRightColumn(Argument<?> argumentSettings)
504                {
505                        return firstParser.metaDescriptionInRightColumn(argumentSettings);
506                }
507
508                @Override
509                String metaDescription(Argument<?> argumentSettings)
510                {
511                        return firstParser.metaDescription(argumentSettings);
512                }
513
514                @Override
515                F defaultValue()
516                {
517                        return transformer.apply(firstParser.defaultValue());
518                }
519
520                @Override
521                ParameterArity parameterArity()
522                {
523                        return firstParser.parameterArity();
524                }
525        }

527        /**
528         * <pre>
529         * Makes it possible to convert several (or zero) {@link String}s into a single {@code T} value.
530         * For a simpler one use {@link StringParser}.
531         *
532         * {@link Argument} is passed to the functions that produces text for the usage,
533         * it can't be a member of this class because one parser can be referenced
534         * from multiple different {@link Argument}s so this is extrinsic state.
535         * </pre>
536         *
537         * @param <T> the type this parser parses strings into
538         */
539        abstract static class InternalStringParser<T>
540        {
541                /**
542                 * @param arguments the arguments given from the command line where
543                 *            {@link ArgumentIterator#next()} points to the parameter
544                 *            for a named argument, for an indexed argument it points to the single unnamed
545                 *            argument
546                 * @param previousOccurance the previously parsed value for this
547                 *            argument if it appears several times, otherwise null
548                 * @param argumentSettings argument settings for this parser,
549                 *            can be used for providing good exception messages.
550                 * @param locale the locale to parse strings with, may matter when parsing numbers, dates
551                 *            etc
552                 * @return the parsed value
553                 * @throws ArgumentException if an error occurred while parsing the value
554                 */
555                abstract T parse(final ArgumentIterator arguments, @Nullable final T previousOccurance, final Argument<?> argumentSettings, Locale locale)
556                                throws ArgumentException;
557
558                /**
559                 * Describes the values this parser accepts
560                 *
561                 * @return a description string to show in usage texts
562                 */
563                @Nonnull
564                abstract String descriptionOfValidValues(Argument<?> argumentSettings, Locale locale);
565
566                /**
567                 * If you can provide a suitable value do so, it will look much better
568                 * in the usage texts and providing sane defaults makes your program/code easier to use,
569                 * otherwise return {@code null}
570                 */
571                @Nullable
572                abstract T defaultValue();
573
574                /**
575                 * Returns a description of {@code value}, or null if no description is
576                 * needed
577                 */
578                @Nullable
579                String describeValue(@Nullable T value)
580                {
581                        return String.valueOf(value);
582                }
583
584                @Nonnull
585                abstract String metaDescription(Argument<?> argumentSettings);
586
587                String metaDescriptionInLeftColumn(Argument<?> argumentSettings)
588                {
589                        return metaDescription(argumentSettings);
590                }
591
592                String metaDescriptionInRightColumn(Argument<?> argumentSettings)
593                {
594                        return metaDescription(argumentSettings);
595                }
596
597                /**
598                 * Returns the {@link ParameterArity}
599                 * {@link #parse(ArgumentIterator, Object, Argument, Locale) parse} expects.
600                 */
601                ParameterArity parameterArity()
602                {
603                        return ParameterArity.AT_LEAST_ONE_ARGUMENT;
604                }
605        }

736        /**
737         * Base class for {@link StringParser}s that uses a sub parser to parse element values and puts
738         * them into a {@link List}
739         */
740        abstract static class ListParser<T> extends InternalStringParser<List<T>>
741        {
742                private final InternalStringParser<T> elementParser;
743
744                private ListParser(final InternalStringParser<T> elementParser)
745                {
746                        this.elementParser = elementParser;
747                }
748
749                protected final InternalStringParser<T> elementParser()
750                {
751                        return elementParser;
752                }
753
754                @Override
755                public String descriptionOfValidValues(Argument<?> argumentSettings, Locale locale)
756                {
757                        return elementParser.descriptionOfValidValues(argumentSettings, locale);
758                }
759
760                @Override
761                String metaDescription(Argument<?> argumentSettings)
762                {
763                        return elementParser.metaDescription(argumentSettings);
764                }
765
766                @Override
767                public List<T> defaultValue()
768                {
769                        return emptyList();
770                }
771
772                @Override
773                String describeValue(List<T> value)
774                {
775                        if(value == null)
776                                return "null";
777                        if(value.isEmpty())
778                                return "Empty list";
779
780                        Iterator<T> values = value.iterator();
781                        String firstValue = String.valueOf(values.next());
782
783                        StringBuilder sb = StringBuilders.withExpectedSize(value.size() * firstValue.length());
784
785                        sb.append(firstValue);
786                        while(values.hasNext())
787                        {
788                                sb.append(", ").append(String.valueOf(values.next()));
789                        }
790                        return sb.toString();
791                }
792        }

794        /**
795         * Implements {@link ArgumentBuilder#arity(int)}
796         */
797        static final class FixedArityParser<T> extends ListParser<T>
798        {
799                private final int arity;
800
801                FixedArityParser(final InternalStringParser<T> parser, final int arity)
802                {
803                        super(parser);
804                        this.arity = arity;
805                }
806
807                @Override
808                List<T> parse(final ArgumentIterator arguments, final List<T> list, final Argument<?> argumentSettings, Locale locale)
809                                throws ArgumentException
810                {
811                        List<T> parsedArguments = new ArrayList<>(arity);
812                        for(int i = 0; i < arity; i++)
813                        {
814                                try
815                                {
816                                        T parsedValue = elementParser().parse(arguments, null, argumentSettings, locale);
817                                        parsedArguments.add(parsedValue);
818                                }
819                                catch(MissingParameterException exception)
820                                {
821                                        // Wrap exception to more clearly specify which parameter that is missing
822                                        throw forMissingNthParameter(exception, i);
823                                }
824                        }
825                        return parsedArguments;
826                }
827
828                @Override
829                public List<T> defaultValue()
830                {
831                        T defaultValue = elementParser().defaultValue();
832                        List<T> listFilledWithDefaultValues = new ArrayList<T>(arity);
833                        for(int i = 0; i < arity; i++)
834                        {
835                                listFilledWithDefaultValues.add(defaultValue);
836                        }
837                        return listFilledWithDefaultValues;

886         * Implements {@link ArgumentBuilder#splitWith(String)}.
887         *
888         * @param <T> the type that's separated by the {@code valueSeparator}
889         */
890        static final class StringSplitterParser<T> extends ListParser<T>
891        {
892                @Nonnull private final String valueSeparator;
893                @Nonnull private final Pattern splitter;
894
895                StringSplitterParser(final String valueSeparator, final InternalStringParser<T> parser)
896                {
897                        super(parser);
898                        check(!valueSeparator.isEmpty(), "Splitting with an empty string is not supported");
899                        this.valueSeparator = valueSeparator;
900                        this.splitter = Pattern.compile(valueSeparator);
901                }
902
903                @Override
904                List<T> parse(final ArgumentIterator arguments, final List<T> oldValue, final Argument<?> argumentSettings, Locale locale)
905                                throws ArgumentException
906                {
907                        if(!arguments.hasNext())
908                                throw forMissingParameter(argumentSettings);
909
910                        String values = arguments.next();
911                        List<T> result = new ArrayList<T>();
912
913                        for(String value : splitter.split(values))
914                        {
915                                value = value.trim();
916                                if(value.isEmpty())
917                                {
918                                        continue;
919                                }
920
921                                ArgumentIterator argument = ArgumentIterator.forArguments(Arrays.asList(value));
922                                T parsedValue = elementParser().parse(argument, null, argumentSettings, locale);
923                                result.add(parsedValue);
924                        }
925                        return result;
926                }
927
928                @Override
929                String metaDescriptionInLeftColumn(Argument<?> argumentSettings)
930                {
931                        String metaDescriptionForValue = metaDescription(argumentSettings);
932                        return metaDescriptionForValue + valueSeparator + metaDescriptionForValue + valueSeparator + "...";
933                }
934        }

965        /**
966         * Implements {@link ArgumentBuilder#asPropertyMap()} and
967         * {@link ArgumentBuilder#asKeyValuesWithKeyParser(StringParser)}.
968         *
969         * @param <K> the type of key in the resulting map
970         * @param <V> the type of values in the resulting map
971         */
972        static final class KeyValueParser<K, V> extends InternalStringParser<Map<K, V>>
973        {
974                @Nonnull private final InternalStringParser<V> valueParser;
975                @Nonnull private final StringParser<K> keyParser;
976                @Nonnull private final Predicate<? super V> valueLimiter;
977                @Nonnull private final Supplier<? extends Map<K, V>> defaultMap;
978
979                KeyValueParser(StringParser<K> keyParser, InternalStringParser<V> valueParser, Predicate<? super V> valueLimiter,
980                                @Nullable Supplier<? extends Map<K, V>> defaultMap, @Nullable final Supplier<? extends V> defaultValue)
981                {
982                        this.valueParser = valueParser;
983                        this.keyParser = keyParser;
984                        this.valueLimiter = valueLimiter;
985                        if(defaultMap == null)
986                        {
987                                this.defaultMap = new Supplier<Map<K, V>>(){
988                                        @Override
989                                        public Map<K, V> get()
990                                        {
991                                                if(defaultValue != null)
992                                                        return new LinkedHashMap<K, V>(){
993                                                                private static final long serialVersionUID = 1L;
994
995                                                                @Override
996                                                                public V get(Object key)
997                                                                {
998                                                                        if(super.containsKey(key))
999                                                                                return super.get(key);
1000                                                                        return defaultValue.get();
1001                                                                }
1002                                                        };
1003                                                return new LinkedHashMap<>();
1004                                        }
1005                                };
1006                        }
1007                        else
1008                        {
1009                                this.defaultMap = defaultMap;
1010                        }
1011                }
1012
1013                @Override
1014                Map<K, V> parse(final ArgumentIterator arguments, Map<K, V> previousMap, final Argument<?> argumentSettings, Locale locale)
1015                                throws ArgumentException
1016                {
1017                        Map<K, V> map = previousMap;
1018                        if(map == null)
1019                        {
1020                                map = defaultValue();
1021                        }
1022
1023                        String keyValue = arguments.next();
1024                        String key = getKey(keyValue, argumentSettings);
1025                        K parsedKey = keyParser.parse(key, locale);
1026                        V oldValue = map.get(parsedKey);
1027
1028                        // Hide what we just did to the parser that handles the "value"
1029                        if(valueParser.parameterArity() != ParameterArity.NO_ARGUMENTS)
1030                        {
1031                                arguments.setNextArgumentTo(getValue(key, keyValue, argumentSettings));
1032                        }
1033                        V parsedValue = valueParser.parse(arguments, oldValue, argumentSettings, locale);
1034
1035                        try
1036                        {
1037                                if(!valueLimiter.test(parsedValue))
1038                                        throw withMessage(format(UserErrors.DISALLOWED_PROPERTY_VALUE, parsedKey, parsedValue, valueLimiter));
1039                        }
1040                        catch(IllegalArgumentException e)
1041                        {
1042                                throw wrapException(e);
1043                        }
1044
1045                        map.put(parsedKey, parsedValue);
1046                        return map;
1047                }
1048
1049                /**
1050                 * Fetch "key" from "key=value"
1051                 */
1052                private String getKey(String keyValue, Argument<?> argumentSettings) throws ArgumentException
1053                {
1054                        if(valueParser.parameterArity() == ParameterArity.NO_ARGUMENTS)
1055                                return keyValue;// Consume the whole string as the key as there's no value to parse
1056                        String separator = argumentSettings.separator();
1057                        int keyEndIndex = keyValue.indexOf(separator);
1058                        if(keyEndIndex == -1)
1059                                throw withMessage(format(UserErrors.MISSING_KEY_VALUE_SEPARATOR, argumentSettings, keyValue, separator));
1060                        return keyValue.substring(0, keyEndIndex);
1061                }
1062
1063                /**
1064                 * Fetch "value" from "key=value"
1065                 */
1066                private String getValue(String key, String keyValue, Argument<?> argumentSettings)
1067                {
1068                        return keyValue.substring(key.length() + argumentSettings.separator().length());
1069                }

1071                @Override
1072                public String descriptionOfValidValues(Argument<?> argumentSettings, Locale locale)
1073                {
1074                        String keyMeta = '"' + keyParser.metaDescription() + '"';
1075                        String valueMeta = '"' + valueParser.metaDescription(argumentSettings) + '"';
1076                        String keyDescription = keyParser.descriptionOfValidValues(locale);
1077                        String valueDescription;
1078                        if(valueLimiter != Predicates2.alwaysTrue())
1079                        {
1080                                valueDescription = valueLimiter.toString();
1081                        }
1082                        else
1083                        {
1084                                valueDescription = valueParser.descriptionOfValidValues(argumentSettings, locale);
1085                        }
1086
1087                        return "where " + keyMeta + " is " + keyDescription + " and " + valueMeta + " is " + valueDescription;
1088                }
1089
1090                @Override
1091                public Map<K, V> defaultValue()
1092                {
1093                        return defaultMap.get();
1094                }
1095
1096                @Override
1097                String metaDescription(Argument<?> argumentSettings)
1098                {
1099                        String keyMeta = keyParser.metaDescription();
1100                        String separator = argumentSettings.separator();
1101                        String valueMeta = valueParser.metaDescription(argumentSettings);
1102                        return keyMeta + separator + valueMeta;
1103                }
