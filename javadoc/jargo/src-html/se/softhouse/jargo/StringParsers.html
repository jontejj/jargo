256        /**
257         * <pre>
258         * A parser that uses {@link Enum#valueOf(Class, String)} to {@link StringParser#parse(String, Locale) parse} input strings.
259         * 
260         * <b>Case sensitivity note:</b> First a direct match with the input value in upper case is
261         * made, if that fails a direct match without converting the case is made,
262         * if that also fails an {@link ArgumentException} is thrown. This order of execution is
263         * based on the fact that users typically don't upper case their input while
264         * <nobr><a href="http://www.oracle.com/technetwork/java/javase/documentation/codeconventions-135099.html#367">java naming conventions</a>
265         * recommends upper case for enum constants.</nobr>
266         * </pre>
267         * 
268         * <b>Default value:</b> <code>null</code> is used as {@link StringParser#defaultValue()} and
269         * not any of the enum values in {@code enumToHandle}
270         * 
271         * @param enumToHandle the {@link Class} literal for the {@link Enum} to parse strings into
272         * @return a {@link StringParser} that parses strings into enum values of the type {@code T}
273         */
274        @CheckReturnValue
275        @Nonnull
276        public static <T extends Enum<T>> StringParser<T> enumParser(final Class<T> enumToHandle)
277        {
278                return new EnumParser<T>(enumToHandle);
279        }
280
281        private static final class EnumParser<E extends Enum<E>> implements StringParser<E>
282        {
283                private final Class<E> enumType;
284
285                private EnumParser(final Class<E> enumToHandle)
286                {
287                        enumType = checkNotNull(enumToHandle);
288                }
289
290                @Override
291                public E parse(final String value, final Locale locale) throws ArgumentException
292                {
293                        try
294                        {
295                                return Enum.valueOf(enumType, value.toUpperCase(Locale.US));
296                        }
297                        catch(IllegalArgumentException noEnumFoundWithUpperCase)
298                        {
299                                try
300                                {
301                                        // Try an exact match just in case an enum value doesn't follow java naming
302                                        // guidelines
303                                        return Enum.valueOf(enumType, value);
304                                }
305                                catch(IllegalArgumentException noEnumFoundWithExactMatch)
306                                {
307                                        throw withMessage(format(UserErrors.INVALID_ENUM_VALUE, value, new Object(){
308                                                @Override
309                                                public String toString()
310                                                {
311                                                        // Lazily call this as it's going over all enum values and converting
312                                                        // them to strings
313                                                        return descriptionOfValidValues(locale);
314                                                }
315                                        }), noEnumFoundWithExactMatch);
316                                }
317                        }
318                }
319
320                @Override
321                public String descriptionOfValidValues(Locale locale)
322                {
323                        E[] enumValues = enumType.getEnumConstants();
324                        StringBuilder values = StringBuilders.withExpectedSize(enumValues.length * AVERAGE_ENUM_NAME_LENGTH);
325                        values.append('{');
326                        Joiner.on(" | ").appendTo(values, enumValues);
327                        values.append('}');
328                        return values.toString();
329                }
330
331                private static final int AVERAGE_ENUM_NAME_LENGTH = 10;
332
333                @Override
334                public E defaultValue()
335                {
336                        return null;
337                }
338
339                @Override
340                public String metaDescription()
341                {
342                        return "<" + enumType.getSimpleName() + ">";
343                }
344        }

463         * <pre>
464         * A {@link Function} that uses {@link StringParser#parse(String, Locale) parse} for input {@link String}s.
465         * </pre>
466         * 
467         * For example:
468         * 
469         * <pre class="prettyprint">
470         * <code class="language-java">
471         * List&lt;Integer&gt; result =  Lists.transform(asList("1", "3", "2"), asFunction(integerParser()));
472         * </code>
473         * </pre>
474         * 
475         * <b>Note:</b>This method may be removed in the future if Guava is removed as a dependency.<br>
476         * <b>Note:</b>The parser will pass the {@link Locale#getDefault()} to
477         * {@link StringParser#parse(String, Locale) parse}. Use
478         * {@link StringParsers#asFunction(StringParser, Locale)} to specify a specific {@link Locale}
479         * 
480         * @param parser the parser to expose as a {@link Function}
481         * @return {@code parser} exposed in a {@link Function} that throws {@link ArgumentException}
482         *         when given faulty values.
483         */
484        @Nonnull
485        @CheckReturnValue
486        @Beta
487        public static <T> Function<String, T> asFunction(final StringParser<T> parser)
488        {
489                return asFunction(parser, Locale.getDefault());
490        }

492        /**
493         * {@link Locale} version of {@link #asFunction(StringParser)}
494         */
495        @Nonnull
496        @CheckReturnValue
497        @Beta
498        public static <T> Function<String, T> asFunction(final StringParser<T> parser, final Locale localeToUse)
499        {
500                checkNotNull(parser);
501                checkNotNull(localeToUse);
502                return new Function<String, T>(){
503                        @Override
504                        public T apply(@Nonnull String input)
505                        {
506                                return parser.parse(input, localeToUse);
507                        }
508                };
509        }

511        /**
512         * <pre>
513         * Makes it possible to convert several (or zero) {@link String}s into a single {@code T} value.
514         * For a simpler one use {@link StringParser}.
515         * 
516         * {@link Argument} is passed to the functions that produces text for the usage,
517         * it can't be a member of this class because one parser can be referenced
518         * from multiple different {@link Argument}s so this is extrinsic state.
519         * 
520         * @param <T> the type this parser parses strings into
521         * </pre>
522         */
523        abstract static class InternalStringParser<T>
524        {
525                /**
526                 * @param arguments the arguments given from the command line where
527                 *            {@link ArgumentIterator#next()} points to the parameter
528                 *            for a named argument, for an indexed argument it points to the single unnamed
529                 *            argument
530                 * @param previousOccurance the previously parsed value for this
531                 *            argument if it appears several times, otherwise null
532                 * @param argumentSettings argument settings for this parser,
533                 *            can be used for providing good exception messages.
534                 * @param locale the locale to parse strings with, may matter when parsing numbers, dates
535                 *            etc
536                 * @return the parsed value
537                 * @throws ArgumentException if an error occurred while parsing the value
538                 */
539                abstract T parse(final ArgumentIterator arguments, @Nullable final T previousOccurance, final Argument<?> argumentSettings, Locale locale)
540                                throws ArgumentException;
541
542                /**
543                 * Describes the values this parser accepts
544                 * 
545                 * @return a description string to show in usage texts
546                 */
547                @Nonnull
548                abstract String descriptionOfValidValues(Argument<?> argumentSettings, Locale locale);
549
550                /**
551                 * If you can provide a suitable value do so, it will look much better
552                 * in the usage texts and providing sane defaults makes your program/code easier to use,
553                 * otherwise return {@code null}
554                 */
555                @Nullable
556                abstract T defaultValue();
557
558                /**
559                 * Returns a description of {@code value}, or null if no description is
560                 * needed
561                 */
562                @Nullable
563                String describeValue(@Nullable T value)
564                {
565                        return String.valueOf(value);
566                }
567
568                @Nonnull
569                abstract String metaDescription(Argument<?> argumentSettings);
570
571                String metaDescriptionInLeftColumn(Argument<?> argumentSettings)
572                {
573                        return metaDescription(argumentSettings);
574                }
575
576                String metaDescriptionInRightColumn(Argument<?> argumentSettings)
577                {
578                        return metaDescription(argumentSettings);
579                }
580
581                /**
582                 * Returns the {@link ParameterArity}
583                 * {@link #parse(ArgumentIterator, Object, Argument, Locale) parse} expects.
584                 */
585                ParameterArity parameterArity()
586                {
587                        return ParameterArity.AT_LEAST_ONE_ARGUMENT;
588                }
589        }

720        /**
721         * Base class for {@link StringParser}s that uses a sub parser to parse element values and puts
722         * them into a {@link List}
723         */
724        abstract static class ListParser<T> extends InternalStringParser<List<T>>
725        {
726                private final InternalStringParser<T> elementParser;
727
728                private ListParser(final InternalStringParser<T> elementParser)
729                {
730                        this.elementParser = elementParser;
731                }
732
733                protected final InternalStringParser<T> elementParser()
734                {
735                        return elementParser;
736                }
737
738                @Override
739                public String descriptionOfValidValues(Argument<?> argumentSettings, Locale locale)
740                {
741                        return elementParser.descriptionOfValidValues(argumentSettings, locale);
742                }
743
744                @Override
745                String metaDescription(Argument<?> argumentSettings)
746                {
747                        return elementParser.metaDescription(argumentSettings);
748                }
749
750                @Override
751                public List<T> defaultValue()
752                {
753                        return emptyList();
754                }
755
756                @Override
757                String describeValue(List<T> value)
758                {
759                        if(value == null)
760                                return "null";
761                        if(value.isEmpty())
762                                return "Empty list";
763
764                        Iterator<T> values = value.iterator();
765                        String firstValue = String.valueOf(values.next());
766
767                        StringBuilder sb = StringBuilders.withExpectedSize(value.size() * firstValue.length());
768
769                        sb.append(firstValue);
770                        while(values.hasNext())
771                        {
772                                sb.append(", ").append(String.valueOf(values.next()));
773                        }
774                        return sb.toString();
775                }
776        }

778        /**
779         * Implements {@link ArgumentBuilder#arity(int)}
780         */
781        static final class FixedArityParser<T> extends ListParser<T>
782        {
783                private final int arity;
784
785                FixedArityParser(final InternalStringParser<T> parser, final int arity)
786                {
787                        super(parser);
788                        this.arity = arity;
789                }
790
791                @Override
792                List<T> parse(final ArgumentIterator arguments, final List<T> list, final Argument<?> argumentSettings, Locale locale)
793                                throws ArgumentException
794                {
795                        List<T> parsedArguments = newArrayListWithCapacity(arity);
796                        for(int i = 0; i < arity; i++)
797                        {
798                                try
799                                {
800                                        T parsedValue = elementParser().parse(arguments, null, argumentSettings, locale);
801                                        parsedArguments.add(parsedValue);
802                                }
803                                catch(MissingParameterException exception)
804                                {
805                                        // Wrap exception to more clearly specify which parameter that is missing
806                                        throw forMissingNthParameter(exception, i);
807                                }
808                        }
809                        return parsedArguments;
810                }
811
812                @Override
813                public List<T> defaultValue()
814                {
815                        T defaultValue = elementParser().defaultValue();
816                        List<T> listFilledWithDefaultValues = newArrayListWithCapacity(arity);
817                        for(int i = 0; i < arity; i++)
818                        {
819                                listFilledWithDefaultValues.add(defaultValue);
820                        }
821                        return listFilledWithDefaultValues;
822                }
823
824                @Override
825                String metaDescriptionInLeftColumn(Argument<?> argumentSettings)
826                {

942        /**
943         * Implements {@link ArgumentBuilder#asPropertyMap()} &
944         * {@link ArgumentBuilder#asKeyValuesWithKeyParser(StringParser)}.
945         * 
946         * @param <K> the type of key in the resulting map
947         * @param <V> the type of values in the resulting map
948         */
949        @VisibleForTesting
950        static final class KeyValueParser<K, V> extends InternalStringParser<Map<K, V>>
951        {
952                @Nonnull private final InternalStringParser<V> valueParser;
953                @Nonnull private final StringParser<K> keyParser;
954                @Nonnull private final Predicate<? super V> valueLimiter;
955                @Nonnull private final Supplier<? extends Map<K, V>> defaultMap;
956
957                KeyValueParser(StringParser<K> keyParser, InternalStringParser<V> valueParser, Predicate<? super V> valueLimiter,
958                                @Nullable Supplier<? extends Map<K, V>> defaultMap, @Nullable final Supplier<? extends V> defaultValue)
959                {
960                        this.valueParser = valueParser;
961                        this.keyParser = keyParser;
962                        this.valueLimiter = valueLimiter;
963                        if(defaultMap == null)
964                        {
965                                this.defaultMap = new Supplier<Map<K, V>>(){
966                                        @Override
967                                        public Map<K, V> get()
968                                        {
969                                                if(defaultValue != null)
970                                                        return new LinkedHashMap<K, V>(){
971                                                                private static final long serialVersionUID = 1L;
972
973                                                                @Override
974                                                                public V get(Object key)
975                                                                {
976                                                                        if(super.containsKey(key))
977                                                                                return super.get(key);
978                                                                        return defaultValue.get();
979                                                                }
980                                                        };
981                                                return Maps.newLinkedHashMap();
982                                        }
983                                };
984                        }
985                        else
986                        {
987                                this.defaultMap = defaultMap;
988                        }
989                }
990
991                @Override
992                Map<K, V> parse(final ArgumentIterator arguments, Map<K, V> previousMap, final Argument<?> argumentSettings, Locale locale)
993                                throws ArgumentException
994                {
995                        Map<K, V> map = previousMap;
996                        if(map == null)
997                        {
998                                map = defaultValue();
999                        }
1000
1001                        String keyValue = arguments.next();
1002                        String key = getKey(keyValue, argumentSettings);
1003                        K parsedKey = keyParser.parse(key, locale);
1004                        V oldValue = map.get(parsedKey);
1005
1006                        // Hide what we just did to the parser that handles the "value"
1007                        if(valueParser.parameterArity() != ParameterArity.NO_ARGUMENTS)
1008                        {
1009                                arguments.setNextArgumentTo(getValue(key, keyValue, argumentSettings));
1010                        }
1011                        V parsedValue = valueParser.parse(arguments, oldValue, argumentSettings, locale);
1012
1013                        try
1014                        {
1015                                if(!valueLimiter.apply(parsedValue))
1016                                        throw withMessage(format(UserErrors.DISALLOWED_PROPERTY_VALUE, parsedKey, parsedValue, valueLimiter));
1017                        }
1018                        catch(IllegalArgumentException e)
1019                        {
1020                                throw wrapException(e);
1021                        }
1022
1023                        map.put(parsedKey, parsedValue);
1024                        return map;
1025                }
1026
1027                /**
1028                 * Fetch "key" from "key=value"
1029                 */
1030                private String getKey(String keyValue, Argument<?> argumentSettings) throws ArgumentException
1031                {
1032                        if(valueParser.parameterArity() == ParameterArity.NO_ARGUMENTS)
1033                                return keyValue;// Consume the whole string as the key as there's no value to parse
1034                        String separator = argumentSettings.separator();
1035                        int keyEndIndex = keyValue.indexOf(separator);
1036                        if(keyEndIndex == -1)
1037                                throw withMessage(format(UserErrors.MISSING_KEY_VALUE_SEPARATOR, argumentSettings, keyValue, separator));
1038                        return keyValue.substring(0, keyEndIndex);
1039                }
1040
1041                /**
1042                 * Fetch "value" from "key=value"
1043                 */
1044                private String getValue(String key, String keyValue, Argument<?> argumentSettings)
1045                {
1046                        return keyValue.substring(key.length() + argumentSettings.separator().length());
1047                }

1049                @Override
1050                public String descriptionOfValidValues(Argument<?> argumentSettings, Locale locale)
1051                {
1052                        String keyMeta = '"' + keyParser.metaDescription() + '"';
1053                        String valueMeta = '"' + valueParser.metaDescription(argumentSettings) + '"';
1054                        String keyDescription = keyParser.descriptionOfValidValues(locale);
1055                        String valueDescription;
1056                        if(valueLimiter != Predicates.alwaysTrue())
1057                        {
1058                                valueDescription = valueLimiter.toString();
1059                        }
1060                        else
1061                        {
1062                                valueDescription = valueParser.descriptionOfValidValues(argumentSettings, locale);
1063                        }
1064
1065                        return "where " + keyMeta + " is " + keyDescription + " and " + valueMeta + " is " + valueDescription;
1066                }
1067
1068                @Override
1069                public Map<K, V> defaultValue()
1070                {
1071                        return defaultMap.get();
1072                }

1074                @Override
1075                String metaDescription(Argument<?> argumentSettings)
1076                {
1077                        String keyMeta = keyParser.metaDescription();
1078                        String separator = argumentSettings.separator();
1079                        String valueMeta = valueParser.metaDescription(argumentSettings);
1080                        return keyMeta + separator + valueMeta;
1081                }
