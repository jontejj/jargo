013package se.softhouse.jargo;
014
015import static java.util.Objects.requireNonNull;
016import static java.util.stream.Collectors.toList;
017import static se.softhouse.common.guavaextensions.Preconditions2.check;
018import static se.softhouse.common.guavaextensions.Predicates2.in;
019import static se.softhouse.jargo.Argument.IS_INDEXED;
020import static se.softhouse.jargo.Argument.IS_REQUIRED;
021
022import java.util.Collection;
023import java.util.HashSet;
024import java.util.LinkedHashMap;
025import java.util.Locale;
026import java.util.Map;
027import java.util.Optional;
028import java.util.Set;
029import java.util.stream.Stream;
030
031import javax.annotation.CheckReturnValue;
032import javax.annotation.Nonnull;
033import javax.annotation.Nullable;
034import javax.annotation.concurrent.Immutable;
035
036import se.softhouse.jargo.internal.Texts.ProgrammaticErrors;
037
038/**
039 * Holds parsed arguments for a {@link CommandLineParser#parse(String...)} invocation.
040 * Use {@link #get(Argument)} to fetch a parsed command line value.
041 */
042@Immutable
043public final class ParsedArguments
044{
045        /**
046         * Stores results from {@link StringParser#parse(String, Locale)}
047         */
048        @Nonnull private final Map<Argument<?>, Object> parsedArguments = new LinkedHashMap<>(4);
049        @Nonnull private final Set<Argument<?>> allArguments;
050
051        /**
052         * This gives commands access to any args to the root/parent command arguments
053         */
054        @Nonnull private final Optional<ParsedArguments> parent;
055
056        /**
057         * The parser that created this instance
058         */
059        @Nonnull private final CommandLineParserInstance source;
060
061        /**
062         * Keeps a running total of how many indexed arguments that have been parsed
063         */
064        private int indexedArgumentsParsed = 0;
065
066        ParsedArguments(CommandLineParserInstance source, ParsedArguments parent)
067        {
068                this.allArguments = source.allArguments();
069                this.source = source;
070                this.parent = Optional.of(parent);
071        }
072
073        ParsedArguments(CommandLineParserInstance source)
074        {
075                this.allArguments = source.allArguments();
076                this.source = source;
077                this.parent = Optional.empty();
078        }
079
080        /**
081         * Returns the parsed value for the given {@code argumentToFetch},
082         * if no value was given on the command line the default value is
083         * returned.
084         * 
085         * @see StringParser#defaultValue()
086         */
087        @Nullable
088        @CheckReturnValue
089        public <T> T get(final Argument<T> argumentToFetch)
090        {
091                if(!wasGiven(argumentToFetch))
092                {
093                        check(handlesArgument(argumentToFetch), ProgrammaticErrors.ILLEGAL_ARGUMENT, argumentToFetch);
094                        return argumentToFetch.defaultValue();
095                }
096                return getValue(argumentToFetch);
097        }
098
099        /**
100         * Checks if an {@link Argument} was given in the command line arguments
101         * 
102         * @param argument the {@link Argument} to check
103         * @return true if {@code argument} was given, that is, the default value will not be used
104         */
105        public boolean wasGiven(Argument<?> argument)
106        {
107                return parsedArguments.containsKey(requireNonNull(argument)) || parent.map(args -> args.wasGiven(argument)).orElse(false);
108        }
109
110        private boolean handlesArgument(Argument<?> arg)
111        {
112                return allArguments.contains(arg) || parent.map(args -> args.handlesArgument(arg)).orElse(false);
113        }
114
115        @Override
116        public String toString()
117        {
118                return parsedArguments.toString();
119        }
120
121        @Override
122        public boolean equals(@Nullable Object other)
123        {
124                if(this == other)
125                        return true;
126                if(!(other instanceof ParsedArguments))
127                        return false;
128
129                return parsedArguments.equals(((ParsedArguments) other).parsedArguments);
130        }
131
132        @Override
133        public int hashCode()
134        {
135                return parsedArguments.hashCode();
136        }
137
138        // Publicly this class is Immutable, CommandLineParserInstance is only allowed to modify it
139        // during parsing
140        @SuppressWarnings("unchecked")
141        <T> T put(final Argument<T> definition, @Nullable final T value)
142        {
143                return (T) parsedArguments.put(definition, value);
144        }
145
146        /**
147         * Makes the parsed value for {@code definition} unmodifiable.
148         */
149        <T> void finalize(final Argument<T> definition)
150        {
151                T value = getValue(definition);
152                T finalizedValue = definition.finalizer().apply(value);
153                put(definition, finalizedValue);
154        }
155
156        <T> T getValue(final Argument<T> definition)
157        {
158                if(parsedArguments.containsKey(definition))
159                {
160                        // Safe because put guarantees that the map is heterogeneous
161                        @SuppressWarnings("unchecked")
162                        T value = (T) parsedArguments.get(definition);
163                        return value;
164                }
165                return parent.map(args -> args.getValue(definition)).orElse(null);
166        }
167
168        Collection<Argument<?>> requiredArgumentsLeft()
169        {
170                return allArguments.stream().filter(IS_REQUIRED.and(in(parsedArguments.keySet()).negate())).collect(toList());
171        }
172
173        boolean hasNonIndexedRequiredArgumentsLeft()
174        {
175                return allArguments.stream().filter(IS_REQUIRED.and(IS_INDEXED.negate()).and(in(parsedArguments.keySet()).negate())).findFirst().isPresent();
176        }
177
178        int indexedArgumentsParsed()
179        {
180                return indexedArgumentsParsed;
181        }
182
183        void incrementIndexedArgumentsParsed()
184        {
185                indexedArgumentsParsed++;
186        }
187
188        Set<Argument<?>> parsedArguments()
189        {
190                return parsedArguments.keySet();
191        }
192
193        Stream<Argument<?>> allArgumentsRecursively()
194        {
195                return Stream.concat(allArguments.stream(), parent.map(r -> r.allArgumentsRecursively()).orElse(Stream.empty()));
196        }
197
198        Set<String> nonParsedArguments()
199        {
200                Set<String> validArguments = new HashSet<>(allArguments.size());
201                allArgumentsRecursively().forEach(argument -> {
202                        boolean wasGiven = wasGiven(argument);
203                        if(!wasGiven || argument.isAllowedToRepeat())
204                        {
205                                for(String name : argument.names())
206                                {
207                                        if(argument.isPropertyMap())
208                                        {
209                                                validArguments.add(name);
210                                        }
211                                        else
212                                        {
213                                                validArguments.add(name + argument.separator());
214                                        }
215
216                                }
217                        }
218                });
219                return validArguments;
220        }
221
222        CommandLineParserInstance parser()
223        {
224                return source;
225        }
226
227        Optional<ParsedArguments> parentHolder()
228        {
229                return parent;
230        }
231
232        CommandLineParserInstance rootParser()
233        {
234                if(parent.isPresent())
235                        return parent.get().rootParser();
236                return parser();
237        }
238
239        Optional<ParsedArguments> findParentHolderFor(Argument<ParsedArguments> argument)
240        {
241                if(allArguments.contains(argument))
242                        return Optional.of(this);
243                else if(parent.isPresent())
244                        return parent.get().findParentHolderFor(argument);
245                return Optional.empty();
246        }
247}
