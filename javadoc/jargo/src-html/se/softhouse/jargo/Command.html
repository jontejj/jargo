013package se.softhouse.jargo;
014
015import static java.util.Arrays.asList;
016import static java.util.Collections.unmodifiableList;
017import static se.softhouse.jargo.Arguments.command;
018import static se.softhouse.jargo.CommandLineParser.US_BY_DEFAULT;
019
020import java.util.ArrayList;
021import java.util.Collections;
022import java.util.List;
023import java.util.Locale;
024import java.util.function.Supplier;
025
026import javax.annotation.CheckReturnValue;
027import javax.annotation.Nonnull;
028import javax.annotation.concurrent.Immutable;
029
030import se.softhouse.common.guavaextensions.Suppliers2;
031import se.softhouse.common.strings.Describable;
032import se.softhouse.jargo.StringParsers.InternalStringParser;
033import se.softhouse.jargo.internal.Texts.UsageTexts;
034
035/**
036 * <pre>
037 * {@link Command}s automatically gets an invocation of execute when given on the command line.
038 * This is particularly useful to avoid a never ending switch statement.
039 *
040 * {@link Command}s have a {@link CommandLineParser} themselves (and thereby sub-commands are allowed as well), that is,
041 * they execute a command and may support contextual arguments as specified by the constructor {@link Command#Command(Argument...)}.
042 *
043 * Sub-commands are executed before their parent {@link Command}.
044 *
045 * To integrate your {@link Command} into an {@link Argument} use {@link Arguments#command(Command)}
046 * or {@link CommandLineParser#withCommands(Command...)} if you have several commands.
047 *
048 * If you support several commands and a user enters several of them at the same
049 * time they will be executed in the order given to {@link CommandLineParser#parse(String...)}.
050 * If any {@link StringParser#parse(String, Locale) parse} errors occurs (for {@link Command#Command(Argument...) command arguments}) no {@link Command}
051 * will be executed. So all arguments, for all commands given, are parsed before any execution occurs.
052 *
053 * <b>Mutability note:</b> although a {@link Command} should be {@link Immutable}
054 * the objects it handles doesn't have to be. So repeated invocations of execute
055 * is allowed to yield different results or to affect external state.
056 * </pre>
057 *
058 * Example subclass:
059 *
060 * <pre class="prettyprint">
061 * <code class="language-java">
062 * public class Build extends Command
063 * {
064 *   private static final Argument&lt;File&gt; PATH = Arguments.fileArgument().description("the directory to build").build();
065 *
066 *   public Build()
067 *   {
068 *     super(PATH);
069 *   }
070 *
071 *   public String description()
072 *   {
073 *     return "Builds a target";
074 *   }
075 *
076 *   protected void execute(ParsedArguments parsedArguments)
077 *   {
078 *     File pathToBuild = parsedArguments.get(PATH);
079 *     //Build pathToBuild here
080 *   }
081 * }
082 * </code>
083 * </pre>
084 *
085 * And the glue needed to integrate the Build {@link Command} with a {@link CommandLineParser}:
086 *
087 * <pre class="prettyprint">
088 * <code class="language-java">
089 * CommandLineParser.withCommands(new Build()).parse("build", "some_directory_that_needs_building");
090 * </code>
091 * </pre>
092 *
093 * As can be seen in the example the command name is the class name in lower case by default. This
094 * may be overridden with {@link #commandName()}.<br>
095 * If your commands don't require any arguments you can use
096 * {@link CommandLineParser#withCommands(Class)} instead.
097 */
098@Immutable
099public abstract class Command extends InternalStringParser<ParsedArguments> implements Describable
100{
101        @Nonnull private final List<Argument<?>> commandArguments;
102
103        @Nonnull private final Supplier<CommandLineParserInstance> commandArgumentParser = Suppliers2.memoize(new Supplier<CommandLineParserInstance>(){
104                @Override
105                public CommandLineParserInstance get()
106                {
107                        return new CommandLineParserInstance(commandArguments, ProgramInformation.AUTO, US_BY_DEFAULT, true);
108                }
109        });
110
111        /**
112         * @param commandArguments the arguments that this command supports.
113         */
114        protected Command(Argument<?> ... commandArguments)
115        {
116                this.commandArguments = unmodifiableList(asList(commandArguments));
117        }
118
119        /**
120         * @param commandArguments the arguments that this command supports.
121         */
122        protected Command(List<Argument<?>> commandArguments)
123        {
124                this.commandArguments = Collections.unmodifiableList(commandArguments);
125        }
126
127        /**
128         * Useful if your {@link Command} has subcommands that you'll want to pass into the
129         * {@link Command#Command(List)} constructor
130         * 
131         * @param commands the subcommands
132         * @return the subcommands as an argument list
133         */
134        public static List<Argument<?>> subCommands(final Command ... commands)
135        {
136                List<Argument<?>> commandsAsArguments = new ArrayList<>(commands.length);
137                for(Command c : commands)
138                {
139                        commandsAsArguments.add(command(c).build());
140                }
141                return commandsAsArguments;
142        }
143
144        /**
145         * The name that triggers this command. Defaults to {@link Class#getSimpleName()} in lower
146         * case. For several names override this with {@link ArgumentBuilder#names(String...)}
147         */
148        @Nonnull
149        @CheckReturnValue
150        protected String commandName()
151        {
152                return getClass().getSimpleName().toLowerCase(Locale.US);
153        }
154
155        /**
156         * Called when this command is encountered on the command line
157         *
158         * @param parsedArguments a container with parsed values for the command arguments,
159         *            as specified by {@link Command#Command(Argument...)}
160         */
161        protected abstract void execute(ParsedArguments parsedArguments);
162
163        /**
164         * Override to provide a description to print in the usage text for this command.
165         * This is essentially an alternative to {@link ArgumentBuilder#description(Describable)}
166         *
167         * @return the description to use in the usage text
168         */
169        @Override
170        @Nonnull
171        public String description()
172        {
173                return "";
174        }
175
176        @Override
177        public String toString()
178        {
179                return commandName();
180        }
181
182        @Override
183        final ParsedArguments parse(final ArgumentIterator arguments, final ParsedArguments previousOccurance, final Argument<?> argumentSettings,
184                        Locale locale) throws ArgumentException
185        {
186                arguments.rememberAsCommand();
187                if(previousOccurance != null)
188                        return resumeParsing(arguments, previousOccurance, locale);
189                ParsedArguments parsedArguments = parser().parse(arguments, locale);
190
191                arguments.rememberInvocationOfCommand(this, parsedArguments, argumentSettings, commandArguments);
192
193                return parsedArguments;
194        }
195
196        final ParsedArguments resumeParsing(final ArgumentIterator arguments, final ParsedArguments previousOccurance, Locale locale)
197                        throws ArgumentException
198        {
199                ParsedArguments result = parser().parseArguments(previousOccurance, arguments, locale);
200                arguments.temporaryRepitionAllowedForCommand = false;
201                return result;
202        }
203
204        /**
205         * The parser for parsing the {@link Argument}s passed to {@link Command#Command(Argument...)}
206         */
207        CommandLineParserInstance parser()
208        {
209                return commandArgumentParser.get();
210        }
211
212        @Override
213        final ParsedArguments defaultValue()
214        {
215                return null;
216        }
217
218        @Override
219        final String descriptionOfValidValues(Argument<?> argumentSettings, Locale locale)
220        {
221                // For commands the validValues is a usage text itself for the command arguments
222                return parser().usage(locale).toString();
223        }
224
225        @Override
226        final String describeValue(ParsedArguments value)
227        {
228                return null;
229        }
230
231        @Override
232        String metaDescription(Argument<?> argumentSettings)
233        {
234                return "";
235        }
236
237        @Override
238        String metaDescriptionInRightColumn(Argument<?> argumentSettings)
239        {
240                return UsageTexts.ARGUMENT_HEADER;
241        }
242}
