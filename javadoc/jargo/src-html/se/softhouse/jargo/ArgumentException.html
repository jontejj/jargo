013package se.softhouse.jargo;
014
015import static java.util.Objects.requireNonNull;
016import static se.softhouse.common.strings.StringsUtil.NEWLINE;
017
018import java.io.Serializable;
019
020import javax.annotation.concurrent.NotThreadSafe;
021
022import se.softhouse.common.strings.StringsUtil;
023import se.softhouse.jargo.internal.Texts.ProgrammaticErrors;
024import se.softhouse.jargo.internal.Texts.UsageTexts;
025
026/**
027 * Indicates that something went wrong in a {@link CommandLineParser#parse(String...) parsing}.
028 * Typical causes include:
029 * <ul>
030 * <li>Missing parameters</li>
031 * <li>Unknown arguments, if it's {@link StringsUtil#closestMatches(String, Iterable, int) close
032 * enough} to a known argument the error message will contain suggestions</li>
033 * <li>Missing required arguments</li>
034 * <li>Invalid arguments, thrown from {@link StringParser#parse(String, java.util.Locale)
035 * parse}</li>
036 * <li>Repetition of argument that hasn't specified {@link ArgumentBuilder#repeated() repeated}</li>
037 * </ul>
038 * The typical remedy action is to present {@link #getMessageAndUsage()} to the user so he is
039 * informed about what he did wrong. <br>
040 * As all {@link Exception}s should be, {@link ArgumentException} is {@link Serializable} so usage
041 * is available after deserialization.
042 */
043@NotThreadSafe
044public abstract class ArgumentException extends RuntimeException
045{
046        /**
047         * The {@link Usage} explaining how to avoid this exception
048         */
049        private Usage usage = null;
050
051        /**
052         * The used name, one of the strings passed to {@link ArgumentBuilder#names(String...)}
053         */
054        private String usedArgumentName = null;
055
056        /**
057         * The first name passed to {@link ArgumentBuilder#names(String...)}. Used to print a reference
058         * to the explanation of the erroneous argument. Used instead of usedArgumentName as the first
059         * name is the leftmost in the listings.
060         */
061        private String usageReferenceName = null;
062
063        protected ArgumentException()
064        {
065        }
066
067        /**
068         * Alias for {@link #initCause(Throwable)}.<br>
069         * Added simply because {@code withMessage("Message").andCause(exception)} flows better.
070         */
071        public final ArgumentException andCause(Throwable cause)
072        {
073                initCause(requireNonNull(cause));
074                return this;
075        }
076
077        /**
078         * Returns usage detailing how to use the {@link CommandLineParser} that caused this
079         * {@link ArgumentException exception}, prepended with an error message detailing the erroneous
080         * argument and, if applicable, a reference to the usage where the user can read about
081         * acceptable input.
082         */
083        public final Usage getMessageAndUsage()
084        {
085                // TODO(jontejj): jack into the uncaughtExceptionHandler and remove stacktraces? Potentially
086                // very annoying feature...
087                String message = getMessage(usedArgumentName) + usageReference() + NEWLINE + NEWLINE;
088                return getUsage().withErrorMessage(message);
089        }
090
091        /**
092         * Returns why this exception occurred.
093         * 
094         * @param referenceName the argument name that was used on the command line.
095         *            If indexed the {@link ArgumentBuilder#metaDescription(String)} is given.
096         *            Furthermore if the argument is both indexed and part of a {@link Command} the
097         *            command name used to trigger the command is given. Alas never null.
098         */
099        protected abstract String getMessage(String referenceName);
100
101        /**
102         * <pre>
103         * {@inheritDoc}
104         * 
105         * Marked as final as the {@link #getMessage(String)} should be implemented instead
106         * </pre>
107         * 
108         * @see #getMessageAndUsage()
109         */
110        @Override
111        public final String getMessage()
112        {
113                return getMessage(usedArgumentName);
114        }
115
116        /**
117         * Returns the usage explaining how to use the {@link CommandLineParser} that
118         * caused this exception.
119         */
120        private Usage getUsage()
121        {
122                requireNonNull(usage, ProgrammaticErrors.NO_USAGE_AVAILABLE);
123                return usage;
124        }
125
126        final ArgumentException withUsage(final Usage theUsage)
127        {
128                if(usage != null)
129                        return this; // Don't overwrite if the user has specified a custom Usage
130                usage = theUsage;
131                return this;
132        }
133
134        final ArgumentException withUsedArgumentName(String argumentNameThatTriggeredMe)
135        {
136                usedArgumentName = argumentNameThatTriggeredMe;
137                return this;
138        }
139
140        /**
141         * By default ". See usage for --author for proper values." is appended to the end of the error
142         * message if {@link #getMessageAndUsage()} is used. This method overrides that to be
143         * {@code usageReference} instead.
144         */
145        void withUsageReference(final String usageReference)
146        {
147                usageReferenceName = requireNonNull(usageReference);
148        }
149
150        ArgumentException withUsageReference(final Argument<?> usageReference)
151        {
152                if(usageReferenceName != null)
153                        return this; // Don't overwrite if the user has specified a custom reference
154                usageReferenceName = String.format(UsageTexts.USAGE_REFERENCE, usageReference);
155                return this;
156        }
157
158        private String usageReference()
159        {
160                return usageReferenceName != null ? usageReferenceName : "";
161        }
162
163        /**
164         * For {@link Serializable}
165         */
166        private static final long serialVersionUID = 1L;
167}
